126//-------------------------------------------------------------------------//
127/* <impl:UIDGen> */
128//-------------------------------------------------------------------------//
129// Do a try catch block around this method if you wish to refresh on overflow.
130UIDGen::Iter UIDGen::GetId() {
131 if (next_id_ == SIZE_MAX) throw std::overflow_error("UIDGen: ID overflow");
132 size_t id = next_id_;
133 next_id_++;
134 if (generated_ids_.contains(id))
135 throw std::runtime_error("UIDGen: ID exists");
136 generated_ids_.insert(id);
137 return generated_ids_.find(id);
138}
139
140void UIDGen::PopId() {
141 if (generated_ids_.empty()) throw std::runtime_error("UIDGen: No IDs to pop");
142 generated_ids_.erase(next_id_);
143 next_id_--;
144}
145
146// Erases an id but does not reset counter. If generating many id and deleting
147// use Referesh() to reset the counter and set.
148void UIDGen::EraseId(Iter it) {
149 if (not generated_ids_.empty() and (it == std::prev(generated_ids_.end()))) {
150 // If the id we are erasing is the last id, we can also pop to save make
151 // more uids
152 PopId();
153 } else {
154 if (generated_ids_.contains(*it)) generated_ids_.erase(it);
155 }
156}
157
158const std::unordered_set<size_t>& UIDGen::Generated() { return generated_ids_; }
159
160// Pops all ids and sets next_id_ = 0;
161void UIDGen::Refresh() {
162 generated_ids_.clear();
163 next_id_ = 1;
164}

220class ScopedWidgetBase {
221 public:
222 // Returns true is the Begin() function has been called
223 // further gui commands will add to this scope.
224 constexpr inline bool IsScopeActive() const { return is_scope_active_; }
225
226 // The meaning of is_on_ may vary based on the ImGui widget call.
227 // Usually indicates if this object was rendered by ImGui.
228 // Some Object will always render, while the result represents a state.
229 // These objects must be ended with ForceEndImpl().
230 // Ex. Windows will return false for this parameter if they are minimized,
231 // but the top bar will still be rendered. So the window is NOT displayed
232 // but it IS rendered.
233 constexpr inline bool IsOn() const { return is_on_; }
234
235 // Implementations of the BeginLate and EndEarly methods.
236 // Begin and End implementations are ment for the constructor
237 // and destructor respectivley.
238 // ForceEnd is for widgets which must call End()
239 // even if Begin() returns false.
240 protected:
241 // void BindBegin(const BoundScopeBeginFuncT& bound_begin) {
242 // begin_func = bound_begin;
243 // }
244 // void BindEnd(const BoundScopeEndFuncT& bound_end) { end_func = bound_end; }
245 bool BeginImpl() {
246 if (is_delayed_ == kWidgetInitImmediate) {
247 is_scope_active_ = true;
248 is_on_ = BoundBegin();
249 } else {
250 is_scope_active_ = false;
251 is_on_ = false;
252 }
253 return is_on_;
254 }
255 bool BeginLateImpl() {
256 if (not is_scope_active_) {
257 is_scope_active_ = true;
258 is_on_ = BoundBegin();
259 }
260 return is_on_;
261 }
262
263 void EndImpl() {
264 if (is_scope_active_ && is_on_) BoundEnd();
265 }
266 void EndEarlyImpl() {
267 if (not is_scope_active_) throw "[ EndEarly() called before begin. ]";
268 is_scope_active_ = false;
269 if (is_on_) BoundEnd();
270 };
271 void ForceEndImpl() {
272 if (is_scope_active_) BoundEnd();
273 }
274 void ForceEndEarlyImpl() {
275 if (not is_scope_active_) throw "[ EndEarly() called before begin. ]";
276 is_scope_active_ = false;
277 BoundEnd();
278 };
279
280 // Public interface to be implemented by the child class.
281 public:
282 virtual bool BoundBegin() = 0;
283 virtual void BoundEnd() = 0;
284 virtual bool BeginLate() = 0;
285 virtual void EndEarly() = 0;
286 virtual ~ScopedWidgetBase() = default;
287
288 public:
289 // Default object is a null widget,Begin() will always return false.
290 ScopedWidgetBase(bool is_delayed)
291 : name_map_(gCguiDefaultUniqueNameMap),
292 id_gen_(gCguiDefaultUIDGenerator),
293 is_delayed_(is_delayed) {
294 is_on_ = false;
295 is_scope_active_ = false;
296 };
297
298 // Copy is allowed, but the name_map and id_gen are shared.
299 // The uniqueness of the name or id is only checked on construction.
300 ScopedWidgetBase(const ScopedWidgetBase& other) = default;
301
302 // Move is allowed.
303 ScopedWidgetBase(ScopedWidgetBase&& other) noexcept = default;
304
305 // Boolean conversion, returns IsOn()
306 operator bool() { return is_on_; }
307
308 protected:
309 UniqueNameMap& name_map_;
310 UIDGen& id_gen_;
311 bool is_scope_active_;
312 bool is_on_;
313 bool is_delayed_;
314
315 protected:
316 // Name must not exist.
317 cxx::BoolError RequestNewName(const std::string& str) {
318 if (name_map_.Contains(str))
319 return "[WidgetBase:NewName:This widget name is already in use.]";
320 name_map_.AddName(str);
321 return true;
322 };
323
324 void ReleaseName(const std::string& str) {
325 // Remove the name from name map.
326 name_map_.RemoveName(str);
327 }

334 id_gen_.EraseId(id_iter);

1567 int TextEditCallback(ImGuiInputTextCallbackData* data) {
1568 // AddLog("cursor: %d, selection: %d-%d", data->CursorPos,
1569 // data->SelectionStart, data->SelectionEnd);
1570 switch (data->EventFlag) {
1571 case ImGuiInputTextFlags_CallbackCompletion: {
1572 // Example of TEXT COMPLETION
1573
1574 // Locate beginning of current word
1575 const char* word_end = data->Buf + data->CursorPos;
1576 const char* word_start = word_end;
1577 while (word_start > data->Buf) {
1578 const char c = word_start[-1];
1579 if (c == ' ' || c == '\t' || c == ',' || c == ';') break;
1580 word_start--;
1581 }
1582
1583 // Build a list of candidates
1584 ImVector<const char*> candidates;
1585 for (int i = 0; i < Commands.Size; i++)
1586 if (Strnicmp(Commands[i], word_start, (int)(word_end - word_start)) ==
1587 0)
1588 candidates.push_back(Commands[i]);
1589
1590 if (candidates.Size == 0) {
1591 // No match
1592 AddLog("No match for \"%.*s\"!\n", (int)(word_end - word_start),
1593 word_start);
1594 } else if (candidates.Size == 1) {
1595 // Single match. Delete the beginning of the word and replace it
1596 // entirely so we've got nice casing.
1597 data->DeleteChars((int)(word_start - data->Buf),
1598 (int)(word_end - word_start));
1599 data->InsertChars(data->CursorPos, candidates[0]);
1600 data->InsertChars(data->CursorPos, " ");
1601 } else {
1602 // Multiple matches. Complete as much as we can..
1603 // So inputing "C"+Tab will complete to "CL" then display "CLEAR" and
1604 // "CLASSIFY" as matches.
1605 int match_len = (int)(word_end - word_start);
1606 for (;;) {
1607 int c = 0;
1608 bool all_candidates_matches = true;
1609 for (int i = 0; i < candidates.Size && all_candidates_matches; i++)
1610 if (i == 0)
1611 c = toupper(candidates[i][match_len]);
1612 else if (c == 0 || c != toupper(candidates[i][match_len]))
1613 all_candidates_matches = false;
1614 if (!all_candidates_matches) break;
1615 match_len++;
1616 }
1617
1618 if (match_len > 0) {
1619 data->DeleteChars((int)(word_start - data->Buf),
1620 (int)(word_end - word_start));
1621 data->InsertChars(data->CursorPos, candidates[0],
1622 candidates[0] + match_len);
1623 }
1624
1625 // List matches
1626 AddLog("Possible matches:\n");
1627 for (int i = 0; i < candidates.Size; i++)
1628 AddLog("- %s\n", candidates[i]);
1629 }
1630
1631 break;
1632 }
1633 case ImGuiInputTextFlags_CallbackHistory: {
1634 // Example of HISTORY
1635 const int prev_history_pos = HistoryPos;
1636 if (data->EventKey == ImGuiKey_UpArrow) {
1637 if (HistoryPos == -1)
1638 HistoryPos = History.Size - 1;
1639 else if (HistoryPos > 0)
1640 HistoryPos--;
1641 } else if (data->EventKey == ImGuiKey_DownArrow) {
1642 if (HistoryPos != -1)
1643 if (++HistoryPos >= History.Size) HistoryPos = -1;
1644 }
1645
1646 // A better implementation would preserve the data on the current input
1647 // line along with cursor position.
1648 if (prev_history_pos != HistoryPos) {
1649 const char* history_str =
1650 (HistoryPos >= 0) ? History[HistoryPos] : "";
1651 data->DeleteChars(0, data->BufTextLen);
1652 data->InsertChars(0, history_str);
1653 }
1654 }
1655 }
1656 return 0;
1657 }
