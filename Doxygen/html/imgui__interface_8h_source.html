141///////////////////////////////////////////////////////////////////////////////
142class UIDGen {
143 public:
144 using Iter = std::unordered_set<size_t>::iterator;
145 UIDGen() : next_id_(1) {} // Starts at 1 cause ImGui reserves id 0
146 // Do a try catch block around this method if you wish to refresh on overflow.
147 Iter GetId();
148 void PopId();
149 void EraseId(Iter it);
150 const std::unordered_set<size_t>& Generated();
151 void Refresh();
152 bool Empty() { return generated_ids_.empty(); }
153
154 private:
155 size_t next_id_;
156 std::unordered_set<size_t> generated_ids_;
157};
158
159///////////////////////////////////////////////////////////////////////////////
160/* <impl:UIDGen> */
161///////////////////////////////////////////////////////////////////////////////
162/// Do a try catch block around this method if you wish to refresh on overflow.
163UIDGen::Iter UIDGen::GetId() {
164 if (next_id_ == SIZE_MAX) throw std::overflow_error("UIDGen: ID overflow");
165 size_t id = next_id_;
166 next_id_++;
167 if (generated_ids_.contains(id))
168 throw std::runtime_error("UIDGen: ID exists");
169 generated_ids_.insert(id);
170 return generated_ids_.find(id);
171}
172
173void UIDGen::PopId() {
174 if (generated_ids_.empty()) throw std::runtime_error("UIDGen: No IDs to pop");
175 generated_ids_.erase(next_id_);
176 next_id_--;
177}
178
179/// Erases an id but does not reset counter.
180/// If generating many ids and deleting, use Referesh() to reset the counter.
181void UIDGen::EraseId(Iter it) {
182 if (not generated_ids_.empty() and (it == std::prev(generated_ids_.end()))) {
183 // If the id we are erasing is the last id, we can also pop to save make
184 // more uids
185 PopId();
186 } else {
187 if (generated_ids_.contains(*it)) generated_ids_.erase(it);
188 }
189}
190
191const std::unordered_set<size_t>& UIDGen::Generated() { return generated_ids_; }
192
193/// Pops all ids and sets next_id_ = 0;
194void UIDGen::Refresh() {
195 generated_ids_.clear();
196 next_id_ = 1;
197}

254class ScopedWidgetBase {
255 public:
256 /// Returns true is the Begin() function has been called
257 /// further gui commands will add to this scope.
258 constexpr inline bool IsScopeActive() const { return is_scope_active_; }
259
260 /// The meaning of is_on_ may vary based on the ImGui widget call.
261 /// Usually indicates if this object was rendered by ImGui.
262 /// Some Object will always render, while the result represents a state.
263 /// These objects must be ended with ForceEndImpl().
264 /// Ex. Windows will return false for this parameter if they are minimized,
265 /// but the top bar will still be rendered. So the window is NOT displayed
266 /// but it IS rendered.
267 constexpr inline bool IsOn() const { return is_on_; }
268
269 /// Implementations of the BeginLate and EndEarly methods.
270 /// Begin and End implementations are ment for the constructor
271 /// and destructor respectivley.
272 /// ForceEnd is for widgets which must call End()
273 /// even if Begin() returns false.
274 protected:
275 // void BindBegin(const BoundScopeBeginFuncT& bound_begin) {
276 // begin_func = bound_begin;
277 // }
278 // void BindEnd(const BoundScopeEndFuncT& bound_end) { end_func = bound_end; }
279 bool BeginImpl() {
280 if (is_delayed_ == kWidgetInitImmediate) {
281 is_scope_active_ = true;
282 is_on_ = BoundBegin();
283 } else {
284 is_scope_active_ = false;
285 is_on_ = false;
286 }
287 return is_on_;
288 }
289 bool BeginLateImpl() {
290 if (not is_scope_active_) {
291 is_scope_active_ = true;
292 is_on_ = BoundBegin();
293 }
294 return is_on_;
295 }
296
297 void EndImpl() {
298 if (is_scope_active_ && is_on_) BoundEnd();
299 }
300 void EndEarlyImpl() {
301 if (not is_scope_active_) throw "[ EndEarly() called before begin. ]";
302 is_scope_active_ = false;
303 if (is_on_) BoundEnd();
304 };
305 void ForceEndImpl() {
306 if (is_scope_active_) BoundEnd();
307 }
308 void ForceEndEarlyImpl() {
309 if (not is_scope_active_) throw "[ EndEarly() called before begin. ]";
310 is_scope_active_ = false;
311 BoundEnd();
312 };
313
314 // Public interface to be implemented by the child class.
315 public:
316 virtual bool BoundBegin() = 0;
317 virtual void BoundEnd() = 0;
318 virtual bool BeginLate() = 0;
319 virtual void EndEarly() = 0;
320 virtual ~ScopedWidgetBase() = default;
321
322 public:
323 /// Default object is a null widget,Begin() will always return false.
324 ScopedWidgetBase(bool is_delayed)
325 : name_map_(gCguiDefaultUniqueNameMap),
326 id_gen_(gCguiDefaultUIDGenerator),
327 is_delayed_(is_delayed) {
328 is_on_ = false;
329 is_scope_active_ = false;
330 };
331
332 /// Copy is allowed, but the name_map and id_gen are shared.
333 /// The uniqueness of the name or id is only checked on construction.
334 ScopedWidgetBase(const ScopedWidgetBase& other) = default;
335
336 /// Move is allowed.
337 ScopedWidgetBase(ScopedWidgetBase&& other) noexcept = default;
338
339 /// Boolean conversion, returns IsOn()
340 operator bool() { return is_on_; }
341
342 protected:
343 UniqueNameMap& name_map_;
344 UIDGen& id_gen_;
345 bool is_scope_active_;
346 bool is_on_;
347 bool is_delayed_;
348
349 protected:
350 /// Name must not exist.
351 cxx::BoolError RequestNewName(const std::string& str) {
352 if (name_map_.Contains(str))
353 return "[WidgetBase:NewName:This widget name is already in use.]";
354 name_map_.AddName(str);
355 return true;
356 };
357
358 void ReleaseName(const std::string& str) {
359 /// Remove the name from name map.
360 name_map_.RemoveName(str);
361 }
362
363 /// Generate a new uuid
364 UIDGen::Iter RequestId() { return id_gen_.GetId(); }
365
366 void ReleaseId(UIDGen::Iter id_iter) {
367 /// Remove the id
368 id_gen_.EraseId(id_iter);
369 }
370};

503 this->close_button_state_ = std::move(other.close_button_state_);
504 }
505
506 protected:
507 bool BoundBegin() override {
508 if (has_close_button_) {
509 close_button_state_ = std::make_unique<bool>(true);
510 }
511 return ImGui::Begin(title_.c_str(), close_button_state_.get(),
512 flags_.Get());
513 }
514
515 void BoundEnd() override { ImGui::End(); }
516
517 private:
518 std::string title_{""};
519 WindowFlags flags_{WindowFlags()};
520 bool has_close_button_;
521 std::unique_ptr<bool> close_button_state_{nullptr};
522 CguiVec2 size_{0.f, 0.f};
523};

1599 int TextEditCallback(ImGuiInputTextCallbackData* data) {
1600 // AddLog("cursor: %d, selection: %d-%d", data->CursorPos,
1601 // data->SelectionStart, data->SelectionEnd);
1602 switch (data->EventFlag) {
1603 case ImGuiInputTextFlags_CallbackCompletion: {
1604 // Example of TEXT COMPLETION
1605
1606 // Locate beginning of current word
1607 const char* word_end = data->Buf + data->CursorPos;
1608 const char* word_start = word_end;
1609 while (word_start > data->Buf) {
1610 const char c = word_start[-1];
1611 if (c == ' ' || c == '\t' || c == ',' || c == ';') break;
1612 word_start--;
1613 }
1614
1615 // Build a list of candidates
1616 ImVector<const char*> candidates;
1617 for (int i = 0; i < Commands.Size; i++)
1618 if (Strnicmp(Commands[i], word_start, (int)(word_end - word_start)) ==
1619 0)
1620 candidates.push_back(Commands[i]);
1621
1622 if (candidates.Size == 0) {
1623 // No match
1624 AddLog("No match for \"%.*s\"!\n", (int)(word_end - word_start),
1625 word_start);
1626 } else if (candidates.Size == 1) {
1627 // Single match. Delete the beginning of the word and replace it
1628 // entirely so we've got nice casing.
1629 data->DeleteChars((int)(word_start - data->Buf),
1630 (int)(word_end - word_start));
1631 data->InsertChars(data->CursorPos, candidates[0]);
1632 data->InsertChars(data->CursorPos, " ");
1633 } else {
1634 // Multiple matches. Complete as much as we can..
1635 // So inputing "C"+Tab will complete to "CL" then display "CLEAR" and
1636 // "CLASSIFY" as matches.
1637 int match_len = (int)(word_end - word_start);
1638 for (;;) {
1639 int c = 0;
1640 bool all_candidates_matches = true;
1641 for (int i = 0; i < candidates.Size && all_candidates_matches; i++)
1642 if (i == 0)
1643 c = toupper(candidates[i][match_len]);
1644 else if (c == 0 || c != toupper(candidates[i][match_len]))
1645 all_candidates_matches = false;
1646 if (!all_candidates_matches) break;
1647 match_len++;
1648 }
1649
1650 if (match_len > 0) {
1651 data->DeleteChars((int)(word_start - data->Buf),
1652 (int)(word_end - word_start));
1653 data->InsertChars(data->CursorPos, candidates[0],
1654 candidates[0] + match_len);
1655 }
1656
1657 // List matches
1658 AddLog("Possible matches:\n");
1659 for (int i = 0; i < candidates.Size; i++)
1660 AddLog("- %s\n", candidates[i]);
1661 }
1662
1663 break;
1664 }
1665 case ImGuiInputTextFlags_CallbackHistory: {
1666 // Example of HISTORY
1667 const int prev_history_pos = HistoryPos;
1668 if (data->EventKey == ImGuiKey_UpArrow) {
1669 if (HistoryPos == -1)
1670 HistoryPos = History.Size - 1;
1671 else if (HistoryPos > 0)
1672 HistoryPos--;
1673 } else if (data->EventKey == ImGuiKey_DownArrow) {
1674 if (HistoryPos != -1)
1675 if (++HistoryPos >= History.Size) HistoryPos = -1;
1676 }
1677
1678 // A better implementation would preserve the data on the current input
1679 // line along with cursor position.
1680 if (prev_history_pos != HistoryPos) {
1681 const char* history_str =
1682 (HistoryPos >= 0) ? History[HistoryPos] : "";
1683 data->DeleteChars(0, data->BufTextLen);
1684 data->InsertChars(0, history_str);
1685 }
1686 }
1687 }
1688 return 0;
1689 }
