49/// @brief Concept to check if a type is streamable.
50///
51/// Non-streamable objects are output as their pointer address.
52template <typename T>
53concept iStreamable = requires(std::ostream os, std::decay_t<T> value) {
54 { os << value } -> std::same_as<std::ostream&>;
55};

57/// @brief A template for a test case, the main 'testing' structure.
58///
59/// @warning: DO NOT directly create an instance of this template.
60/// Always use MINITEST macro.
61template <auto TestName, auto TestCaseName, typename TestImpl>
62struct Test {
63 static constexpr auto test_name = TestName();
64 static constexpr auto test_case_name = TestCaseName();
65 static inline const TestImpl test_impl{};
66 bool is_test_passed = true;
67
68 /// @brief Runs this test and logs result to console(if enabled).
69 /// @return bool always true
70 ///
71 /// Define MINITEST_CONFIG_NO_CONSOLE_PRINT before including minitest to
72 /// disable printing to console when running tests.
73 static inline bool Run() {
74#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
75 std::cout << kSeparator << "[Begin Mini Test] " << test_name << " [Case]"
76 << test_case_name << "\n"
77 << kSeparator;
78#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
79 test_impl();
80#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
81 std::cout << kSeparator << "[End Mini Test] " << test_name << " [Case]"
82 << test_case_name << "\n"
83 << kSeparator;
84#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
85 return true;
86 }
87};

105/// DO NOT USE
106static inline bool ExpectTrue(bool b) {
107 if (!b) {
108#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
109 std::cout << "[FAIL] Expected TRUE." << std::endl;
110#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
111 return false;
112 }
113 return true;
114}
115
116/// DO NOT USE
117static inline bool ExpectFalse(bool b) {
118 if (b) {
119#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
120 std::cout << "[FAIL] Expected FALSE." << std::endl;
121#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
122 return false;
123 }
124 return true;
125}
126
127/// DO NOT USE
128template <typename LT, typename RT>
129 requires std::equality_comparable_with<LT, RT>
130static inline bool ExpectEq(LT lhs, RT rhs) {
131 if (lhs == rhs) {
132 return true;
133 } else {
134#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
135 std::cout << "[FAIL] Expected Equality with: ";
136 if constexpr (iStreamable<LT> && iStreamable<RT>)
137 std::cout << ">|" << lhs << " Got: " << rhs << std::endl;
138 else
139 std::cout << ">|" << &lhs << " Got: " << &rhs << std::endl;
140#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
141 return false;
142 }
143}
144
145/// DO NOT USE
146template <typename LT, typename RT>
147 requires std::equality_comparable_with<LT, RT>
148static inline bool ExpectNe(LT lhs, RT rhs) {
149 if (lhs != rhs) {
150 return true;
151 } else {
152#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
153 std::cout << "[FAIL] Expected Inequality with";
154 if constexpr (iStreamable<LT> && iStreamable<RT>)
155 std::cout << ">|" << lhs << std::endl;
156 else
157 std::cout << ">|" << &lhs << std::endl;
158#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
159 return false;
160 }
161}
162
163/// DO NOT USE
164static inline bool ExpectAnyThrow(auto&& f) {
165 try {
166 f();
167 } catch (...) {
168 return true;
169 }
170#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
171 std::cout << "[FAIL] Expected exception but got none." << std::endl;
172#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
173 return false;
174}
175
176/// DO NOT USE
177static inline bool ExpectNoThrow(auto&& f) {
178 try {
179 f();
180 } catch (...) {
181#ifndef MINITEST_CONFIG_NO_CONSOLE_PRINT
182 std::cout << "[FAIL] Expected no exception but one was raised."
183 << std::endl;
184#endif // !MINITEST_CONFIG_NO_CONSOLE_PRINT
185 return false;
186 }
187 return true;
188}

192/// @brief Stores results and metadata of a recorded test.
193struct TestResult {
194 bool is_test_passed;
195 std::string test_name;
196 std::string test_case_name;
197 std::source_location location;
198 std::string log;
199};
200
201/// @brief QOL std::ostream overload to directly print test results.
202std::ostream& operator<<(std::ostream& ss, const TestResult& t) {
203 if (t.is_test_passed) {
204 ss << "[PASSED] Test: " << t.test_name << "\n Case: " << t.test_case_name
205 << '\n';
206 } else {
207 ss << "[FAILED] Test: " << t.test_name << "\n Case: " << t.test_case_name
208 << t.location.file_name() << '(' << t.location.line() << ':'
209 << t.location.column() << ") `" << t.location.function_name() << "`: \n";
210 };
211 return ss;
212}

214static std::vector<std::string> gFailedTestLogs; ///! Stores failed test logs.
215static std::vector<TestResult> gRecordedTestLogs; ///! Stores all test results if enabled.

217/// Stores the function pointers to a modules test cases associated with a registered name.
218static std::map<std::string_view, std::vector<std::function<bool()>>> gRegisteredTests;
219static const char* gLastFailedTestName = ""; ///! DO NOT USE, impl detail.
220static const char* gLastFailedTestCaseName = ""; ///! DO NOT USE, impl detail.
221
222/// @brief Adds a test case to a test module with given test_name
223/// @param test_name Name of test or test module.
224/// @param test Function pointer to the test.
225static inline void RegisterTest(const std::string_view & test_name,
226 std::function<bool()> test) {
227 gRegisteredTests[test_name].push_back(test);
228}
229
230/// @brief Runs all the tests associated with a given test name or module name.
231/// @param test_name Name of the module to run, must be registered.
232/// @return True, unless a test failed - then false.
233static inline bool RunRegisteredTestModule(const std::string_view & test_name) {
234 bool passed = true;
235 for (const auto& test : gRegisteredTests[test_name]) {
236 if(not test()){
237 passed = false;
238 }
239 }
240 return passed;
241};

243/// @brief For internal mini-test use only. Adds a formatted error log to the
244/// gFailedTestLogs
245/// @param log Message to log.
246/// @param test Test name.
247/// @param tcase Test case name.
248static inline void AddFailedTestLog(
249 const std::string& log, const char* test, const char* tcase,
250 const std::source_location location = std::source_location::current()) {
251 std::stringstream ss;
252 ss << "[FAILURE DETECTED] Test: " << std::string(test)
253 << " Case: " << std::string(tcase) << " On Check:" << log
254 << "\nfile: " << location.file_name() << '(' << location.line() << ':'
255 << location.column() << ") `" << location.function_name() << "`: " << '\n';
256 gFailedTestLogs.push_back(ss.str());
257}
258
259/// @brief Prints all failed test logs, or "All Tests Passed!" when none.
260/// @return True if all tests passed, false otherwise.
261static inline bool PrintFailedTestLogs() {
262 if (gFailedTestLogs.empty()) {
263 std::cout << kSeparator << "All tests passed.\n" << kSeparator;
264 return true;
265 } else {
266 std::cout << kSeparator << "Failed Tests:\n" << kSeparator;
267 for (const auto& log : gFailedTestLogs) {
268 std::cout << log << std::endl;
269 }
270 std::cout << kSeparator << "End of Failed Tests:\n" << kSeparator;
271 return false;
272 }
273}

275/// @brief Adds a TestResult to 'gRecordedTestLogs'
276static inline void RecordTestLog(
277 bool passed, const std::string& log, const char* test, const char* tcase,
278 const std::source_location location = std::source_location::current()) {
279 gRecordedTestLogs.push_back(TestResult{.is_test_passed = passed,
280 .test_name = std::string(test),
281 .test_case_name = std::string(tcase),
282 .location = location,
283 .log = log});
284}
285
286/// @brief Prints all recorded test logs or "No tests were run" if none.
287/// @return Returns true if no tests were ran, false otherwise.
288static inline bool PrintRecordedTestLogs() {
289 if (gRecordedTestLogs.empty()) {
290 std::cout << kSeparator << "No test were run.\n" << kSeparator;
291 return true;
292 } else {
293 std::cout << kSeparator << "Tests:\n" << kSeparator;
294 for (const auto& log : gRecordedTestLogs) {
295 std::cout << log << std::endl;
296 }
297 std::cout << kSeparator << "End of Tests:\n" << kSeparator;
298 return false;
299 }
300}

328/// @def MINITEST
329/// @brief Defines a static test case. Always close with 'END_MINITEST;'.
330/// @param TestName Name of the test.
331/// @param TestCaseName Name of the test case,must be unique per test.
332///
333/// { } before and after the test case definition is optional
334/// but recommended for visibility.
335#define MINITEST(TestName, TestCaseName)
336 namespace minitest_unit_test {
337 namespace TestName {
338 bool MINITEST_TEST_##TestCaseName = []() -> bool {
339 return minitest::Test < []() consteval -> const char* { return #TestName; },
340 []() consteval -> const char* { return #TestCaseName; },
341 decltype([]() -> void {
342 minitest::gLastFailedTestName = #TestName;
343 minitest::gLastFailedTestCaseName = #TestCaseName;
344
345/// @def END_MINITEST
346/// @brief Completes and runs a test case.
347///
348/// Must be called after your test case definition.
349#define END_MINITEST
350 }
351 )
352 >
353 ::Run();
354 }
355 ();
356 }
357 ;
358 }

360/// @def MINITEST_F}
361/// @brief Defines a fixture test case.
362/// Always close with 'END_MINITEST_F(NameOfFixture);'.
363/// @param TestName Name of the test.
364/// @param TestCaseName Name of the test case,must be unique per test.
365/// @param TestFixtureClass Name of the fixture class.
366///
367/// Must inherit from minitest::Fixture.
368/// @see minitest::Fixture
369#define MINITEST_F(TestName, TestCaseName, TestFixtureClass)
370 namespace minitest_unit_test {
371 namespace TestName {
372 bool MINITEST_TEST_##TestCaseName = []() -> bool {
373 return minitest::Test < []() consteval -> const char* { return #TestName; },
374 []() consteval -> const char* { return #TestCaseName; },
375 decltype([]() -> void {
376 const static struct MINITEST_FIXTURE_##TestCaseName : TestFixtureClass {
377 void RunFixture() {
378 minitest::gLastFailedTestCaseName = #TestName;
379 minitest::gLastFailedTestName = #TestCaseName;
380 (*this).SetUp();
381
382
383/// @def END_MINITEST_F
384/// @brief Closes a fixture test case. ONLY use after MINITEST_F with the SAME
385/// TestCaseName.
386/// @param TestCaseName : Name of the last-called test case which created
387/// was created with MINITEST_F. MUST be the same as the last-called
388/// test case. Or else it will not compile.
389///
390/// ex. If you called MINITEST_F(MyTest,MyTestCase,MyFixture)
391/// then you must call END_MINITEST_F(MyTestCase);
392#define END_MINITEST_F(TestCaseName)
393 (*this).TearDown();
394 }
395 }
396 ;
397 MINITEST_FIXTURE_##TestCaseName MINITEST_FIXTURE_INSTANCE_##TestCaseName;
398 MINITEST_FIXTURE_INSTANCE_##TestCaseName.RunFixture();
399 }
400 )
401 >::Run();
402 }
403 ();
404 }
405 ;
406 }

480/// @def MINITEST_INTERNAL_CHECK_METHOD
481/// @brief Generates a checking macro for a given minitest method.
482/// DO NOT USE DIRECTLY.
483/// This is used internally by the EXPECT_ macros.
484#ifdef MINITEST_CONFIG_RECORD_ALL
485#define MINITEST_INTERNAL_CHECK_METHOD(method, pmsg, msg, ...)
486 if (!minitest::method(__VA_ARGS__)) {
487 minitest::AddFailedTestLog(msg, minitest::gLastFailedTestName,
488 minitest::gLastFailedTestCaseName);
489 minitest::RecordTestLog(false, msg, minitest::gLastFailedTestName,
490 minitest::gLastFailedTestCaseName);
491 } else {
492 minitest::RecordTestLog(true, pmsg, minitest::gLastFailedTestName,
493 minitest::gLastFailedTestCaseName);
494 }
495#else
496#define MINITEST_INTERNAL_CHECK_METHOD(method, pmsg, msg, ...)
497 if (!minitest::method(__VA_ARGS__)) {
498 minitest::AddFailedTestLog(msg, minitest::gLastFailedTestName,
499 minitest::gLastFailedTestCaseName);
500 }
501#endif
502
503/// @def MINITEST_INTERNAL_ASSERT_METHOD
504/// @brief Generates am assertion macro for a given minitest method.
505/// DO NOT USE DIRECTLY.
506/// This is used internally by the ASSERT_ macros.
507#ifdef MINITEST_CONFIG_RECORD_ALL
508#define MINITEST_INTERNAL_ASSERT_METHOD(method, pmsg, msg, ...)
509 if (!minitest::method(__VA_ARGS__)) {
510 minitest::AddFailedTestLog(msg, minitest::gLastFailedTestName,
511 minitest::gLastFailedTestCaseName);
512 minitest::RecordTestLog(false, msg, minitest::gLastFailedTestName,
513 minitest::gLastFailedTestCaseName);
514 return;
515 } else {
516 minitest::RecordTestLog(true, pmsg, minitest::gLastFailedTestName,
517 minitest::gLastFailedTestCaseName);
518 }
519#else
520#define MINITEST_INTERNAL_ASSERT_METHOD(method, pmsg, msg, ...)
521 if (!minitest::method(__VA_ARGS__)) {
522 minitest::AddFailedTestLog(msg, minitest::gLastFailedTestName,
523 minitest::gLastFailedTestCaseName);
524 return;
525 }
