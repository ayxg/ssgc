129// Create an undefined named local variable in a given namespace.
130class DeclareLocalVar : public IrActionBase {
131 string_view name{""};
132
133 DeclareLocalVar(RtStaticAllocator &static_memory,
134 RtOperationalAllocator &active_memory,
135 const string_view &name)
136 : IrActionBase(static_memory, active_memory), name(name) {}
137
138 void Perform() override {
139 if (namespace_.ContainsLocal(name)) {
140 error_code_ = eRtErrorCode::kVarAlreadyDeclared;
141 return;
142 }
143 // Cant add directly, memory is owned by static mem.
144 // Must create a new val, add to memory then add to ns.
145 RtStaticAllocator::memory_descriptor new_val = static_memory_.AllocStatic();
146 RtVal &val_ref = new_val.Deref();
147 val_ref = RtVal::NewUndefined();
148 namespace_.AddDecl(name, val_ref);
149 error_code_ = eRtErrorCode::kNoError;
150 }
151};

153// Resolve a local value and push a ref to it onto the hot stack.
154class ResolveLocalVar : public IrActionBase {
155 ResolveLocalVar(RtStaticAllocator &static_memory,
156 RtOperationalAllocator &active_memory,
157 const string_view &name)
158 : IrActionBase(static_memory, active_memory), name(name) {}
159
160 void Perform() override {
161 // Check if the variable is declared.
162 auto var = namespace_.Resolve(name);
163 if (var) {
164 // Push a reference to the variable onto the hot stack.
165 active_memory_.Push(RtDynPtr(var.Extract()).Get());
166 } else {
167 error_code_ = eRtErrorCode::kVarNotDeclared;
168 }
169 }
170
171 private:
172 const string_view name;
173};

175// Assing an RtVal from the top of the working stack to a variable.
176class AssignLocalVar : public IrActionBase {
177 AssignLocalVar(RtStaticAllocator &static_memory,
178 RtOperationalAllocator &active_memory, const string_view &name)
179 : IrActionBase(static_memory, active_memory), name(name) {}
180
181 void Perform() override {
182 // Check if the variable is declared.
183 auto var = namespace_.Resolve(name);
184 if (var) {
185 // Assign the value to the variable.
186 *var.Extract() = active_memory_.PopGet();
187 } else {
188 error_code_ = eRtErrorCode::kVarNotDeclared;
189 }
190 }
191
192 private:
193 const string_view name;
194};

196// Unary operation is always in-place
197class UnaryNegation : public IrActionBase {
198 UnaryNegation(RtStaticAllocator &static_memory,
199 RtOperationalAllocator &active_memory)
200 : IrActionBase(static_memory, active_memory) {}
201
202 void Perform() override {
203 active_memory_.UnaryOperation([this](RtVal &v) { return Visit(v); });
204 }
205
206 inline void Visit(RtVal &val) {
207 error_code_ = eRtErrorCode::kNoError;
208 std::visit(
209 [this](auto &v) {
210 USING_DECAYED_T(v);
211 CXIF(iOneOfNativeLiterals<T>) {
212 CXIF(iOneOfNativeArithmetics<T>) {
213 CXIF(mta::req::iSameAs<T, NativeBoolT>) {
214 // Boolean negation uses the ! operator.
215 v = !v;
216 }
217 else if (mta::req::iSameAs<T, NativeNoneT>) {
218 // Do nothing for a None object.
219 }
220 else {
221 // Its some numeric literal, use - operator.
222 v = -v;
223 }
224 }
225 else if (mta::req::iSameAs<T, NativePtrT>) {
226 // Visit the pointed to val if not a nullptr.
227 if (v.IsNull())
228 error_code_ = eRtErrorCode::kNullPointerOperation;
229 else
230 Visit(v.Get());
231 }
232 else {
233 // Undefined or deleted val. Cannot perform operation.
234 error_code_ = eRtErrorCode::kUndefinedValue;
235 }
236 }
237 else {
238 // Objects should search for the defined member method '_negate_'
239
240 error_code_ = eRtErrorCode::kNotImplemented;
241 }
242 },
243 val.Variant());
244 }
245};

247class BinaryAddition : public IrActionBase {
248 BinaryAddition(RtStaticAllocator &static_memory,
249 RtOperationalAllocator &active_memory)
250 : IrActionBase(static_memory, active_memory) {}
251
252 void Perform() override {
253 active_memory_.BinaryOperation([this](RtVal &&lhs, RtVal &&rhs) {
254 return Visit(forward<RtVal>(lhs), forward<RtVal>(rhs));
255 });
256 }
257
258 template <typename LhsT>
259 requires iOneOfNativeArithmetic<LhsT>
260 inline RtVal &&VisitRhs(LhsT &&lhsv, RtVal &&rhsv) {
261 return std::visit(
262 [&lhsv](auto &&v) {
263 using RhsT = std::decay_t<decltype(rhsv)>;
264 // RHS is a native literal.
265 CXIF(iOneOfNativeLiterals<RhsT>) {
266 // RHS is an arithmetic literal.
267 CXIF(iOneOfNativeArithmetics<RhsT>) {
268 CXIF(mta::req::iSameAs<RhsT, NativeNoneT>) {
269 // Adding none to anything equals anything.
270 return lhsv;
271 }
272 // Both values are native arithmetics.
273 else return lhsv + rhsv;
274 }
275 // RHS is a pointer.
276 else CXIF(mta::req::iSameAs<T, NativePtrT>) {
277 if (v.IsNull()) {
278 error_code_ = eRtErrorCode::kNullPointerOperation;
279 return RtVal::NewUndefined();
280 } else
281 return VisitRhs(forward<LhsT>(lhsv), v.Get());
282 }
283 // RHS is undefined
284 else {
285 error_code_ = eRtErrorCode::kUndefinedValue;
286 return RtVal::NewUndefined();
287 }
288 }
289 },
290 rhs_val.Variant());
291 }
292
293 // Return type is decided based on the type of the lhs
294 // operand.
295 inline RtVal &&Visit(RtVal &&lhs_val, RtVal &&rhs_val) {
296 error_code_ = eRtErrorCode::kNoError;
297 return std::visit(
298 [this, &rhs_val](auto &&v) {
299 using LhsT = std::decay_t<decltype(v)>;
300 // LHS is a native literal.
301 CXIF(iOneOfNativeLiterals<LhsT>) {
302 // LHS is a None object, do nothing.
303 CXIF(mta::req::iSameAs<LhsT, NativeNoneT>) { return v; }
304 // LHS is a pointer to some RtVal.
305 // If non-null pointer, visit the pointed to val.
306 else CXIF(mta::req::iSameAs<LhsT, NativePtrT>) {
307 if (v.IsNull()) {
308 error_code_ = eRtErrorCode::kNullPointerOperation;
309 return RtVal::NewUndefined();
310 } else
311 return Visit(v.Get());
312 }
313 // LHS is an Undefined val.
314 // Cannot perform operation.
315 else CXIF(mta::req::iSameAs<LhsT, NativeUndefinedT>) {
316 error_code_ = eRtErrorCode::kUndefinedValue;
317 return RtVal::NewUndefined();
318 }
319 // Arithmetic literal, visit RHS.
320 else {
321 return VisitRhs<LhsT>(v, rhs_val);
322 }
323 }
324 // LHS is a non-literal.
325 else {
326 // LHS is a string.
327
328 // LHS is a dynamic object
329
330 // LHS is a dynamic list of RtVals
331
332 error_code_ = eRtErrorCode::kNotImplemented;
333 return RtVal::NewUndefined();
334 }
335 },
336 lhs_val.Variant());
337 }
338};
