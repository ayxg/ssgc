689Expected<LitChar> LitChar::FromAst(const Ast& ast) {
690 using namespace caerr;
691 if (ast.TypeIsnt(eAst::LitChar)) {
692 return Expected<LitChar>::Failure("Expected LiteralChar");
693 }
694 // literal will be in the form [0-255]c or '[character]'c
695 if (ast.Literal().back() != 'c') {
696 return Expected<LitChar>::Failure(
697 ("COctetEval:Octet literal not followed by 'c'" + ast.Literal()));
698 }
699 LitChar ret{};
700 // if the literal is a character
701 if (ast.Literal()[0] == '\'') {
702 auto byte_str = ast.Literal().substr(
703 1, ast.Literal().size() - 3); // remove the quotes and the c
704 // Check for escape characters.
705 auto escape_pos = byte_str.find('\\');
706 if (escape_pos != string::npos) {
707 // Replace the escape characters
708 switch (ast.Literal()[escape_pos + 1]) {
709 case 'n':
710 byte_str.replace(escape_pos, 2, "\n");
711 break;
712 case 't':
713 byte_str.replace(escape_pos, 2, "\t");
714 break;
715 case 'r':
716 byte_str.replace(escape_pos, 2, "\r");
717 break;
718 case '0':
719 byte_str.replace(escape_pos, 2, "\0");
720 break;
721 case '\\':
722 byte_str.replace(escape_pos, 2, "\\");
723 break;
724 case '\'':
725 byte_str.replace(escape_pos, 2, "'");
726 break;
727 case '\"':
728 byte_str.replace(escape_pos, 2, "\"");
729 break;
730 default:
731 Expected<LitByte>::Failure("COctetEval:Invalid escape character:" +
732 ast.Literal());
733 }
734 }
735 if (byte_str.size() != 1) {
736 Expected<LitByte>::Failure("COctetEval:Invalid character literal:" +
737 ast.Literal());
738 }
739 ret.val = static_cast<char>(byte_str.at(0));
740 } else { // Has to be a number
741 auto char_str = ast.Literal().substr(0, ast.Literal().size() - 1);
742 ret.val = static_cast<char>(std::stoi(char_str));
743 }
744 return Expected<LitChar>::Success(std::move(ret));
745}

747Expected<LitCstr> LitCstr::FromAst(const Ast& ast) {
748 using namespace caerr;
749 if (ast.TypeIsnt(eAst::LitCstr)) {
750 return Expected<LitCstr>::Failure("Expected LiteralCstr");
751 }
752 // Get the cstr of the node
753 string literal = ast.Literal();
754
755 // Remove the quotes
756 literal = literal.substr(1, literal.size() - 2);
757
758 // Check for escape characters.
759 // If there are escape characters, we need to replace them with their actual
760 // values.
761 bool no_escape = false;
762 size_t current_pos = 0;
763 while (!no_escape) {
764 auto escape_pos = literal.find('\\', current_pos);
765 if (escape_pos == string::npos) {
766 no_escape = true;

840Expected<Program> Program::FromAst(const Ast& ast) {
841 // 1. Root ast must be a program.
842 if (ast.TypeIsnt(eAst::Program)) {
843 return Expected<Program>::Failure("Root ast must be a program.");
844 }
845
846 // 2. Create a base program definition:
847 // - global namespace
848 // - predefined methods vars namespaces can go here.
849 Program program;
850 program.global_ns.eval_order = 0;
851 program.global_ns.name = Ident{kGlobalNamespaceName.data()};
852
853 // 3. Iterate over all the statements and generate IrAst for each.
854 size_t eval_order = 0;
855 for (const auto& stmt : ast.Children()) {
856 // All statements must be pragmatic at the top level.
857 if (not stmt.IsPragmatic()) {
858 return Expected<Program>::Failure(
859 "All statements must be pragmatic at the top level.");
860 }
861 // Generate IrAst for the statements.
862 // Keep track of the order of evaluation
863 if (stmt.TypeIs(eAst::VariableDeclaration) ||
864 stmt.TypeIs(eAst::VariableDefinition)) {
865 auto var_def_ir =
866 VarDecl::FromAst(stmt, program, program.global_ns, eval_order);
867 if (not var_def_ir) return Expected<Program>::Failure(var_def_ir.Error());
868 } else if (stmt.TypeIs(eAst::MethodDeclaration) ||
869 stmt.TypeIs(eAst::MethodDefinition)) {
870 auto fn_def_ir =
871 FnDecl::FromAst(stmt, program, program.global_ns, eval_order);
872 if (not fn_def_ir) return Expected<Program>::Failure(fn_def_ir.Error());
873 } else if (stmt.TypeIs(eAst::KwUse)) {
874 auto use_def_ir =
875 UsingDecl::FromAst(stmt, program, program.global_ns, eval_order);
876 if (not use_def_ir) return Expected<Program>::Failure(use_def_ir.Error());
877 } else if (stmt.TypeIs(eAst::ClassDeclaration)) {
878 auto use_def_ir =
879 UsingDecl::FromAst(stmt, program, program.global_ns, eval_order);
880 if (not use_def_ir) return Expected<Program>::Failure(use_def_ir.Error());
881 } else if (stmt.TypeIs(eAst::KwNamespace)) {
882 auto use_def_ir =
883 UsingDecl::FromAst(stmt, program, program.global_ns, eval_order);
884 if (not use_def_ir) return Expected<Program>::Failure(use_def_ir.Error());
885 }
886 }
887 // 4. Return the program.
888 return Expected<Program>::Success(program);

891Expected<NamespaceDecl*> NamespaceDecl::FromAst(const Ast& ast,
892 Program& program,
893 NamespaceDecl& ns,
894 size_t& eval_order) {
895 if (ast.TypeIsnt(eAst::KwNamespace)) {
896 return Expected<NamespaceDecl*>::Failure("Expected Namespace Declaration");
897 }
898 // Get the namespace name
899 auto ns_name = Ident::FromAst(ast[0]);
900 if (not ns_name) return Expected<NamespaceDecl*>::Failure(ns_name.Error());
901 size_t internal_eval_order = 0;
902 // Check if the namespace is already defined.
903 if (ns.namespaces.contains(ns_name.Value().name)) {
904 auto& existing_ns = ns.namespaces[ns_name.Value().name];
905 // Get the last evaluation order from within that namespace.
906 auto& internal_eval_order = existing_ns.next_evaluation_index;
907 // Add statements to the previously created namespace.
908 for (const auto& stmt : ast.Children()) {
909 if (not stmt.IsPragmatic()) {
910 return Expected<NamespaceDecl*>::Failure(
911 "All statements must be pragmatic at the namespace level.");
912 }
913 if (stmt.TypeIs(eAst::VariableDeclaration) ||
914 stmt.TypeIs(eAst::VariableDefinition)) {
915 auto var_def_ir =
916 VarDecl::FromAst(stmt, program, existing_ns, internal_eval_order);
917 if (not var_def_ir)
918 return Expected<NamespaceDecl*>::Failure(var_def_ir.Error());
919 } else if (stmt.TypeIs(eAst::MethodDeclaration) ||
920 stmt.TypeIs(eAst::MethodDefinition)) {
921 auto fn_def_ir =
922 FnDecl::FromAst(stmt, program, existing_ns, internal_eval_order);
923 if (not fn_def_ir)
924 return Expected<NamespaceDecl*>::Failure(fn_def_ir.Error());
925 } else if (stmt.TypeIs(eAst::KwUse)) {
926 auto use_def_ir =
927 UsingDecl::FromAst(stmt, program, existing_ns, internal_eval_order);
928 if (not use_def_ir)
929 return Expected<NamespaceDecl*>::Failure(use_def_ir.Error());
930 } else if (stmt.TypeIs(eAst::ClassDeclaration)) {
931 auto use_def_ir =
932 UsingDecl::FromAst(stmt, program, existing_ns, internal_eval_order);
933 if (not use_def_ir)
934 return Expected<NamespaceDecl*>::Failure(use_def_ir.Error());
935 } else if (stmt.TypeIs(eAst::KwNamespace)) {
936 auto use_def_ir =
937 UsingDecl::FromAst(stmt, program, existing_ns, internal_eval_order);
938 if (not use_def_ir)
939 return Expected<NamespaceDecl*>::Failure(use_def_ir.Error());
940 }
941 }
942 return Expected<NamespaceDecl*>::Success(&existing_ns);
943 }
944 // Else this is a new namespace.
945 else {
946 // Make sure namespace does not have
947 // the same name as a class, typedef or variable
948 // in the parent namespace.
949 if (ns.classes.contains(ns_name.Value().name)) {
950 return Expected<NamespaceDecl*>::Failure(
951 "Namespace cannot have the same name as a class.");
952 }
953 if (ns.typedefs.contains(ns_name.Value().name)) {
954 return Expected<NamespaceDecl*>::Failure(
955 "Namespace cannot have the same name as a typedef.");
956 }
957 if (ns.fields.contains(ns_name.Value().name)) {
958 return Expected<NamespaceDecl*>::Failure(
959 "Namespace cannot have the same name as a variable.");
960 }
961 if (ns.const_fields.contains(ns_name.Value().name)) {
962 return Expected<NamespaceDecl*>::Failure(
963 "Namespace cannot have the same name as a variable.");
964 }
965 // Create a new namespace and add to parent namespace.
966 ns.namespaces[ns_name.Value().name] = NamespaceDecl{};
967 NamespaceDecl& ns_decl = ns.namespaces[ns_name.Value().name];
968 ns_decl.parent = &ns;
969 ns_decl.eval_order = eval_order;
970 ns_decl.next_evaluation_index = 0;
971 ns_decl.name = ns_name.Extract();
972
973 // Add statements to the newly created namespace.
974 for (const auto& stmt : ast.Children()) {
975 if (not stmt.IsPragmatic()) {
976 return Expected<NamespaceDecl*>::Failure(
977 "All statements must be pragmatic at the namespace level.");
978 }
979 if (stmt.TypeIs(eAst::VariableDeclaration) ||
980 stmt.TypeIs(eAst::VariableDefinition)) {
981 auto var_def_ir =
982 VarDecl::FromAst(stmt, program, ns_decl, internal_eval_order);
983 if (not var_def_ir)
984 return Expected<NamespaceDecl*>::Failure(var_def_ir.Error());
985 } else if (stmt.TypeIs(eAst::MethodDeclaration) ||
986 stmt.TypeIs(eAst::MethodDefinition)) {
987 auto fn_def_ir =
988 FnDecl::FromAst(stmt, program, ns_decl, internal_eval_order);
989 if (not fn_def_ir)
990 return Expected<NamespaceDecl*>::Failure(fn_def_ir.Error());
991 } else if (stmt.TypeIs(eAst::KwUse)) {
992 auto use_def_ir =
993 UsingDecl::FromAst(stmt, program, ns_decl, internal_eval_order);
994 if (not use_def_ir)
995 return Expected<NamespaceDecl*>::Failure(use_def_ir.Error());
996 } else if (stmt.TypeIs(eAst::ClassDeclaration)) {
997 auto use_def_ir =
998 UsingDecl::FromAst(stmt, program, ns_decl, internal_eval_order);
999 if (not use_def_ir)
1000 return Expected<NamespaceDecl*>::Failure(use_def_ir.Error());
1001 } else if (stmt.TypeIs(eAst::KwNamespace)) {
1002 auto use_def_ir =
1003 UsingDecl::FromAst(stmt, program, ns_decl, internal_eval_order);
1004 if (not use_def_ir)
1005 return Expected<NamespaceDecl*>::Failure(use_def_ir.Error());
1006 }
1007 }
1008 // return ptr to created namespace
1009 return Expected<NamespaceDecl*>::Success(&ns_decl);
1010 }

1013Expected<VarDecl*> VarDecl::FromAst(const Ast& ast, Program& program,
1014 NamespaceDecl& context, size_t& eval_order,
1015 eVarDeclCategory decl_category) {
1016 if (not(ast.TypeIs(eAst::VariableDeclaration) ||
1017 ast.TypeIs(eAst::VariableDefinition))) {
1018 return cxx::Expected<VarDecl*>::Failure(
1019 "Root ast must be VariableDeclaration or VariableDefinition");
1020 }
1021
1022 // If this is a static or member variable assert the name is unique
1023 // in the namespace or class or functional or iterative context.
1024 // Iterative variables which collide with names outside of
1025 // the iterative context - up to the owning functional context,
1026 // will emit an error of shadowed identifiers.
1027 // (Warning or Error? Warnings are bad so for now error.)
1028 if (decl_category == eVarDeclCategory::kProgramVar ||
1029 decl_category == eVarDeclCategory::kClassMember) {
1030 auto var_name = Ident::FromAst(ast[2]);
1031 if (not var_name) return Expected<VarDecl*>::Failure(var_name.Error());
1032 if (context.fields.contains(var_name.Value().name) ||
1033 context.const_fields.contains(var_name.Value().name)) {
1034 return Expected<VarDecl*>::Failure(
1035 "Variable name must be unique in the namespace.");
1036 }
1037 }
1038
1039 // 3. Create a new variable definition.
1040 ns.fields[var_name.Value().name] = VarDecl{};
1041 VarDecl& var_def = ns.fields[var_name.Value().name];
1042 var_def.eval_order = eval_order;
1043 var_def.name = var_name.Extract();
1044 var_def.qualified_name = cxx::MergeVectors(ns.qualified_name, {var_def.name});
1045
1046 // 4. Increment eval order.
1047 eval_order++;
1048
1049 // 5. Get the modifiers.
1050 auto mods_ir = ValMods::FromAst(ast[0]);
1051 if (not mods_ir) return mods_ir;
1052 var_def.mods = mods_ir.Extract();
1053
1054 // 6. Type Constraints
1055 auto constraints_ir = TypeList::FromAst(ast[1]);
1056 if (not constraints_ir) return constraints_ir;
1057 var_def.constraints = constraints_ir.Extract();
1058
1059 // 7. Initializer
1060 // If its a var decl with no provided initializer:
1061 // -> Provide a default initializer.
1062 if (ast.TypeIs(eAst::VariableDeclaration)) {
1063 // -> Const ?
1064 if (var_def.mods.is_const) {
1065 return Expected<VarDecl*>::Failure(
1066 "Constant values must be explicitly initialized.");
1067 }
1068 // If the variable is an Owned Value (the default)
1069 // -> If there is a type constraint.
1070 // -> If the type is default initializable provide default init.
1071 // -> Else report an error.
1072 // -> If there is no type constraint and no initializer.
1073 // -> Default initialize to a NoneT
1074 if (var_def.mods.valcat == eValCategory::Owned) {
1075 if (var_def.constraints.IsAny()) {
1076 // Create Default Initializer for NoneT
1077 var_def.initializer = FuncCall(KwNone(), FuncArgs::VoidArgs());
1078 } else {
1079 var_def.initializer =
1080 FuncCall(var_def.constraints.types.at(0), FuncArgs::VoidArgs());
1081 }
1082 }
1083 // If the variable is marked as Reference Value('ref').
1084 // -> If constrained default init to a nullreference of type T
1085 // -> If not constrained default init to a void_reference
1086 else if (var_def.mods.valcat == eValCategory::Reference) {
1087 var_def.initializer = FuncCall(KwVoid(), FuncArgs::VoidArgs());
1088 }
1089 // Is a view?
1090 else if (var_def.mods.valcat == eValCategory::View) {
1091 return Expected<VarDecl*>::Failure(
1092 "Views must always be initialized to a left hand side value.Cannot "
1093 "view void.");
1094 }
1095 } else { // Variable Definition
1096 auto init_expr_ir = PrimaryExpr::FromAst(ast[3], program);
1097 if (not init_expr_ir) return init_expr_ir;
1098 var_def.initializer = init_expr_ir.Extract();
1099 if (var_def.initializer.IsConstEval()) {
1100 var_def.is_consteval = true;
1101 }
1102 }
1103 return Expected<VarDecl*>::Success(&var_def);
1104}
