25/// Represents a boolean true value or an error message.
26class BoolError {
27 public:
28 constexpr bool Valid() const { return valid_; }
29 constexpr const std::string& Error() const { return error_; }
30
31 /// Creates a runtime_error exception out of the Error() string.
32 inline std::runtime_error Exception() const {
33 return std::runtime_error(error_.c_str());
34 }
35
36 /// True if object is in a valid state.
37 constexpr operator bool() const { return valid_; }
38
39 public:
40 BoolError(std::string error_message) : valid_(false), error_(error_message) {}
41 BoolError(const char* error_message) : valid_(false), error_(error_message) {}
42 BoolError() : valid_(true), error_("") {}
43 BoolError(bool value) : valid_(value) {
44 if (value)
45 error_ = "";
46 else
47 error_ = "BoolError: Unspecified false error.";
48 }
49
50 private:
51 /// Is this object in a valid state?
52 bool valid_;
53 /// Error message if in error state.
54 std::string error_;
55};

57/// @brief Represents an expected value or an error message.
58/// @tparam T The value type expected to be held in this object.
59template <typename T>
60class Expected {
61 public:
62 constexpr bool Valid() const { return expected_.has_value(); }
63 constexpr T Extract() {
64 T val = std::move(expected_.value());
65 expected_.reset();
66 error_ = "Value has been moved out of the expected object.";
67 return val;
68 }
69 constexpr const T& Value() const { return expected_.value(); }
70 constexpr const auto& Error() const { return error_; }
71 constexpr operator bool() const { return expected_.has_value(); }
72 static constexpr Expected<T> Success(const T& expected) {
73 return Expected<T>(std::forward<const T>(expected));
74 }
75 static constexpr Expected<T> Success(T&& expected) {
76 return Expected<T>(std::forward<T>(expected));
77 }
78 static constexpr Expected<T> Failure(const std::string& error_message) {
79 return Expected<T>(std::nullopt,
80 std::forward<const std::string>(error_message));
81 }
82 static constexpr Expected<T> Failure(std::string&& error_message) {
83 return Expected<T>(std::nullopt, std::forward<std::string>(error_message));
84 }
85
86 template <typename U>
87 static constexpr Expected<T> NewChainFailure(const Expected<U>& other) {
88 auto ret = Expected<T>::Failure(other.Error());
89 return ret;
90 }
91
92 template <typename U>
93 static constexpr Expected<T> NewChainFailure(
94 const Expected<U>& other, const std::string& error_message) {
95 auto ret = Expected<T>::Failure(other.Error() + "\n" + error_message);
96 return ret;
97 }
98
99 constexpr Expected<T>&& ChainFailure(std::string error_message) {
100 return std::forward<Expected<T>>(
101 Expected<T>::NewChainFailure(*this, error_message));
102 }
103
104 template <typename U>
105 constexpr Expected<T>&& ChainFailure(const Expected<U>& other) {
106 return std::forward<Expected<T>>(Expected<T>::NewChainFailure<U>(other));
107 }
108
109 private:
110 std::optional<T> expected_{std::nullopt};
111 std::string error_{""};
112
113 constexpr Expected(T expected) : expected_(expected) {}
114 template <typename T>
115 constexpr Expected(const T& expected) : expected_(expected) {}
116
117#pragma warning(
118 disable : 4100) // Disable unused parameter warning for std::nullopt_t
119 // Cannot be instantiated directly from type.
120 constexpr Expected(const std::nullopt_t& none, std::string error_message)
121 : expected_(std::nullopt), error_(error_message) {}
122#pragma warning(default : 4100)
123};

125/// @brief PartialExpected returns a value or an error message, but also allows
126/// for an additional value which will always be passed along with the
127/// expected.
128/// @tparam T Expected value type.
129/// @tparam AlwaysT Value type that is always returned.
130template <typename T, typename AlwaysT>
131class PartialExpected {
132 AlwaysT always_;
133 std::optional<T> value_{std::nullopt};
134 std::string error_{""};
135
136 template <typename T, typename AlwaysT>
137 constexpr PartialExpected(const AlwaysT& always, T expected)
138 : always_(always), value_(expected) {}
139
140 template <typename AlwaysT>
141 constexpr PartialExpected(const AlwaysT& always) : always_(always) {}
142
143 public:
144 constexpr bool Valid() const { return value_.has_value(); }
145
146 constexpr const T& Value() const { return value_.value(); }
147
148 constexpr const AlwaysT& Always() const { return always_; }
149
150 constexpr const std::string& Error() const { return error_; }
151 constexpr operator bool() const { return value_.has_value(); }
152 constexpr T Extract() {
153 T val = std::move(value_.value());
154 value_.reset();
155 error_ = "Value has been moved out of the expected object.";
156 return val;
157 }
158
159 constexpr PartialExpected<T, AlwaysT> ChainFailure(
160 std::string error_message) {
161 return PartialExpected::NewChainFailure(*this, error_message);
162 }
163
164 constexpr PartialExpected<T, AlwaysT> ChainFailure() {
165 return PartialExpected::NewChainFailure(*this);
166 }
167
168 static constexpr PartialExpected<T, AlwaysT> Success(const AlwaysT& always,
169 T expected) {
170 return PartialExpected(always, expected);
171 }
172
173 static constexpr PartialExpected<T, AlwaysT> Failure(
174 const AlwaysT& always, const std::string& error_message) {
175 auto ret = PartialExpected(always);
176 ret.error_ = error_message;
177 return ret;
178 }
179
180 static constexpr PartialExpected<T, AlwaysT> NewChainFailure(
181 const PartialExpected& other, const AlwaysT& new_always,
182 const std::string& error_message) {
183 auto ret = PartialExpected::Failure(new_always,
184 other.Error() + "\n" + error_message);
185 return ret;
186 }
187
188 static constexpr PartialExpected<T, AlwaysT> NewChainFailure(
189 const PartialExpected& other, const std::string& error_message) {
190 auto ret = PartialExpected::Failure(other.Always(),

193 }

199 other.Error() + "\n" + error_message);

201 }
202};
