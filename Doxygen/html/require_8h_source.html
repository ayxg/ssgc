333/// <@concept:Numeral_>

339/// <@concept:SpecializedArithmetic_>
340/// <@brief: Specifies an arithmetic type that is specialized in
341/// std::numeric_limits.>

343concept iSpecializedArithmetic =
344 iArithmetic<T> && std::numeric_limits<T>::is_specialized;

346/// <@concept:SpecializedArithmetic_>
347/// <@brief: Specifies an arithmetic type that is specialized and has at least
348/// one value that,
349/// if used as an argument to an arithmetic operation, will generate a
350/// trap.>
351template <typename T>
352concept iTrappingArithmetic =
353 iSpecializedArithmetic<T> && std::numeric_limits<T>::traps;
354
355/// <@concept:Iec559Floating_>
356/// <@brief: Specifies floating-point type which fulfill the requirements of IEC
357/// 559 (IEEE 754) standard. This means that
358/// std::numeric_limits<T>::has_infinity, std::numeric_limits<T>::has_quiet_NaN,
359/// and std::numeric_limits<T>::has_signaling_NaN are also true.>
360template <typename T>
361concept iIec559Floating =
362 iFloatingPoint<T> && std::numeric_limits<T>::is_iec559;
363
364/// <@concept: TrappingFloating_>
365/// <@brief: Specifies an floating type that is specialized and has at least one
366/// value that,
367/// if used as an argument to an arithmetic operation, will generate a
368/// trap.>
369template <typename T>
370concept iTrappingFloating = iFloatingPoint<T> && iTrappingArithmetic<T>;
371
372/// <@concept: FloatingHasInf_>
373/// <@brief: Specifies an floating type that is specialized and capable of

376template <typename T>
377concept iFloatingHasInf =
378 iFloatingPoint<T> && std::numeric_limits<T>::has_infinity;
379
