31static string LoadFileToStr(stdfs::path file_path) {
32 std::filesystem::path filepath(
33 std::filesystem::absolute(std::filesystem::path(file_path)));
34
35 std::uintmax_t fsize;
36
37 if (std::filesystem::exists(filepath)) {
38 fsize = std::filesystem::file_size(filepath);
39 } else {
40 throw(std::invalid_argument("File not found: " + filepath.string()));
41 }
42
43 std::ifstream infile;
44 infile.exceptions(std::ifstream::failbit | std::ifstream::badbit);
45 try {
46 infile.open(filepath.c_str(), std::ios::in | std::ifstream::binary);
47 } catch (...) {
48 std::throw_with_nested(
49 std::runtime_error("Can't open input file " + filepath.string()));
50 }
51
52 string fileStr;
53
54 try {
55 fileStr.resize(fsize);
56 } catch (...) {
57 std::stringstream err;
58 err << "Can't resize to " << fsize << " bytes";
59 std::throw_with_nested(std::runtime_error(err.str()));
60 }
61
62 infile.read(fileStr.data(), fsize);
63 infile.close();
64
65 return fileStr;
66}

524 for (auto& tab : file_tabs) {
525 if (tab.file_path == file_path) {
526 tab.is_active = true;
527 } else {
528 tab.is_active = false;
529 }
530 }
531
532 auto active_tab =
533 std::find_if(file_tabs.begin(), file_tabs.end(),
534 [](const FileTab& tab) { return tab.is_active; });
535
536 if (active_tab == file_tabs.end()) {
537 AddFileTab(file_path);
538 return SetFileTabActive(file_path);
539 } else {
540 active_tab_ = active_tab;

545 // Sets active file tab given an index. if the index is out of bounds
546 // method will return error.

549 return ExpectedFileTabRef::Failure("Index out of bounds.");

562 }
