22/// @def LAMBDA
23///
24/// Convention:
25/// - Use lambda macro instead of auto when declaring lambdas
26/// - Lambda variable should be CamelCase with a preceding 'x'.
27/// 'x' stands for lambda eXpression.
28/// ex. LAMBDA xMyLambda = []() {};
29#define LAMBDA auto

31/// @def LAMBDA_STRING
32/// @brief helper macro to create a lambda that returns a string at compile time.

45/// @def FOR_EACH_256(macro, ...)
46///
47/// Expands __VA_ARGS__ into a functional macro with a limit of 342 times.
48/// (EXPAND4 gets called 256 times, but the intermediary macros cause rescan as
49/// well) Source: Recursive macros with C++20 __VA_OPT__ by David Mazières
50/// https://www.scs.stanford.edu/~dm/blog/va-opt.html
51#define FOR_EACH_256(macro, ...)
52 __VA_OPT__(EXPAND_256(FOR_EACH_HELPER_256(macro, __VA_ARGS__)))
53#define FOR_EACH_HELPER_256(macro, a1, ...)
54 macro(a1) __VA_OPT__(FOR_EACH_AGAIN_256 PARENS_256(macro, __VA_ARGS__))
55#define FOR_EACH_AGAIN_256() FOR_EACH_HELPER_256
56
57#define PARENS_1024 ()

59#define EXPAND_1024(arg)
60 EXPAND1_1024(EXPAND1_1024(EXPAND1_1024(EXPAND1_1024(arg))))
61#define EXPAND1_1024(arg)
62 EXPAND2_1024(EXPAND2_1024(EXPAND2_1024(EXPAND2_1024(arg))))
63#define EXPAND2_1024(arg)
64 EXPAND3_1024(EXPAND3_1024(EXPAND3_1024(EXPAND3_1024(arg))))
65#define EXPAND3_1024(arg)

72/// @see FOR_EACH_256(macro, ...) for details.
