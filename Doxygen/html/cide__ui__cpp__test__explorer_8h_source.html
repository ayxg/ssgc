28 // Structure stored for each callback set added to the test explorer.
29 // The callback does not neccesarily have to be a minitest test case.
30 // Only requirement is that it returns a bool and takes no arguments.
31 class TestModuleResult {
32 friend class CideTestExplorerInterface;
33 public:
34 // Run the scase and store the result in this class.
35 void Run() { is_test_passed_ = test_case_(); }
36
37 bool IsEnabled() const { return *run_this_test; }
38
39 bool IsPassed() const { return is_test_passed_; }
40
41 const string& Name() const { return test_module_name_; }
42
43 // Enable or disable the test case.
44 void SetEnabled(bool value = true) { *run_this_test = value; }
45
46 public:
47 TestModuleResult(const string& test_module_name, bool is_test_passed,
48 BoolCallbackT test_case)
49 : test_module_name_(test_module_name),
50 is_test_passed_(is_test_passed),
51 test_case_(test_case) {}
52 private:
53 //! Only use when required for ImGui Callbacks.
54 bool* GetRawEnableTestPtr() { return run_this_test; }
55
56 bool is_test_passed_{false};
57 BoolCallbackT test_case_{xNullBoolCallback};
58 bool* run_this_test = new (bool)(false);
59 string test_module_name_{""};
60 };

182 void DisplayToolbar() {
183 toolbar_context.RequestSize({window_width, window_height * 0.1f});
184 if (toolbar_context.BeginLate()) {
185 if (run_all_tests_button.BeginLate()) {
186 RunEnabledModuleTests();

190 minitest::FlushTestResults();

192 }
193 toolbar_context.EndEarly();

196 void DisplayModulesTable() {
197 modules_context.RequestSize({window_width * 0.30f, window_height});
198 if (modules_context.BeginLate()) {
199 if (ImGui::BeginTable("test-explorer-modules-table", 2, 1)) {
200 int flags = 0;
201 flags |= ImGuiTableFlags_SizingFixedFit;
202 ImGui::TableSetupColumn("Status", flags);
203 ImGui::TableSetupColumn("Module", flags);
204 ImGui::TableHeadersRow();
205 for (size_t style_id = 0; auto& test_case : registered_test_cases) {
206 if (test_case.IsPassed()) {
207 FillModulesTablePassRowData(style_id, test_case);
208 } else {
209 FillModulesTableFailRowData(style_id, test_case);
210 }
211 }
212 ImGui::EndTable();
213 }
214 }
215 modules_context.EndEarly();
216 }
217
218 void DisplayChecksTable() {
219 test_results_context.RequestSize({window_width * 0.70f, window_height});
220 if (test_results_context.BeginLate()) {
221 if (ImGui::BeginTable("test-explorer-checks-table", 5, 1)) {
222 int flags = 0;
223 flags |= ImGuiTableFlags_SizingFixedFit;
224 ImGui::TableSetupColumn("Status", flags);
225 ImGui::TableSetupColumn("Test", flags);
226 ImGui::TableSetupColumn("Test Case", flags);
227 ImGui::TableSetupColumn("Message", flags);
228 ImGui::TableSetupColumn("Location");
229 ImGui::TableHeadersRow();
230
231 for (size_t style_id = 0; const auto& test_case : MINITESTS_RECORDED) {
232 if (test_case.is_test_passed) {
233 FillChecksTablePassRowData(style_id, test_case);
234 } else {
235 FillChecksTableFailRowData(style_id, test_case);
236 }
237 }
238 ImGui::EndTable();
239 }
240 }
241 test_results_context.EndEarly();
242 }
