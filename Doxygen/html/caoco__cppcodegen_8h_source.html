56_scxin cstring kCommaSeparator = ", ";
57
58_scxin string GenerateExprList(const vector<string>& exprs,
59 const string& separator = kCommaSeparator,
60 bool separator_after_last = false,
61 const string& open_with = "",
62 const string& close_with = "",
63 const string& prefix = "") {
64 string ret = open_with;
65 if (exprs.size() > 1) {
66 for (auto arg_iter = exprs.cbegin(); arg_iter != std::prev(exprs.cend());
67 arg_iter++) {
68 ret += prefix + *arg_iter + separator;
69 }
70 }
71 if (not exprs.empty()) {
72 ret += prefix + exprs.back();
73 }
74 return ret + (separator_after_last ? separator : "") + close_with;
75}

77_scxin string GenerateExprList(
78 const vector<std::reference_wrapper<const string>>& exprs,
79 const string& separator = kCommaSeparator,
80 bool separator_after_last = false, const string& open_with = "",
81 const string& close_with = "", const string& prefix = "") {
82 string ret = open_with;
83 if (exprs.size() > 1) {
84 for (auto arg_iter = exprs.cbegin(); arg_iter != std::prev(exprs.cend());
85 arg_iter++) {
86 ret += prefix + arg_iter->get() + separator;
87 }
88 }
89 if (not exprs.empty()) {
90 ret += prefix + exprs.back().get();
91 }
92 return ret + (separator_after_last ? separator : "") + close_with;

118struct IncludeDirective {
119 enum class IncludeType {
120 Quotes, /// Use quotes for the include directive.
121 AngleBrackets /// Use angle brackets for the include directive.
122 };
123 /// The header to include.
124 string header{};
125 /// The type of include directive
126 /// (Quotes or AngleBrackets).
127 IncludeType type{IncludeType::Quotes};
128
129 /// @brief Code generation method.
130 ///
131 /// @return Generated include directive string.
132 _cx string Codegen() const {
133 if (type == IncludeType::Quotes)
134 return "#include \"" + header + "\"\n";
135 else
136 return "#include <" + header + ">\n";
137 }
138};

140/// Generates a #define macro.
141struct MacroDefine {
142 enum class eMacroType : bool { Definition = true, Functional = false };
143 eMacroType macro_type{eMacroType::Definition};
144 /// Name of the macro.Not checked to be unique.
145 string ident{""};
146 /// Definition to give the macro. Make sure to
147 /// use the '\' character before newlines in the
148 /// macro definition to output correct C++ code.
149 string def{""};
150 vector<string> args{""};
151
152 _cx string Codegen() const {
153 if (macro_type == eMacroType::Definition) {
154 return "#define " + ident + " " + def + "\n";
155 } else {
156 return GenerateExprList(args, kCommaSeparator, false,
157 "#define " + ident + "(", ") " + def + "\n");
158 }
159 }
160};

162/// Represents a template parameter (either a class or a specific type).
163struct TemplateTypeParam {
164 enum class eTemplateType : bool { Type = true, Value = false };
165 /// Name of the template parameter.
166 string name{""};
167 eTemplateType template_type{eTemplateType::Type};
168 /// The type if this is a value template.
169 string type{""};
170 /// True is this is a pack eg. <class...T>
171 bool is_pack{false};
172 _cx string Codegen() const {
173 if (template_type == eTemplateType::Type)
174 if (is_pack)
175 return "class ... " + name;
176 else
177 return "class " + name;
178 else { // eTemplateType::Value
179 if (is_pack)
180 return type + " ... " + name;
181 else
182 return type + " " + name;
183 }
184 }
185};

210/// Generates a variable declaration eg. int foo = 42;
211struct VariableDecl {
212 enum class eInitType {
213 Declaration, /// No initializer will be generated. Variable declaration.
214 BracketInit, /// Init will be placed in brackets. eg. int foo(42);
215 BraceInit, /// Init will be placed in braces. eg. int foo{42};
216 Assignment /// Init will be assigned by '=' operator. eg. int foo = 42;
217 };
218 /// The type of initializer to use for this declaration.
219 eInitType init_type{
220 eInitType::Declaration};
221 string ident{""};
222 string type{""};
223 /// Primary expression to emplace in the
224 /// variable initializer. Based on init type.
225 /// Ignore if init_type is eInitType::Declaration.
226 /// This primary expression must not end with a
227 /// semicolon, codegen will insert a ';' if necessary.
228 string init{
229 ""};
230
231 _cx string Codegen() const {
232 switch (init_type) {
233 case eInitType::Declaration:
234 return type + " " + ident + ";\n";
235 case eInitType::BracketInit:
236 return type + " " + ident + "(" + init + ");\n";
237 case eInitType::BraceInit:
238 return type + " " + ident + "{" + init + "};\n";
239 case eInitType::Assignment:
240 return type + " " + ident + " = " + init + ";\n";
241 default:
242 throw cxx::UnknownEnumEntry<eInitType>();
243 }
244 }
245};

247/// @brief Generates a using declaration eg. using Int = int;
248struct UsingDecl {
249 enum class eTypedefType {
250 Typealias,
251 NamespaceExposition,
252 DeclarationExposition
253 };
254 /// The category typedef.
255 eTypedefType init_type;
256 /// Name of the typedef.
257 string ident{""};
258
259 /// Primary expression to emplace in the type alias.
260 /// Based on init type. Ignore if init_type is
261 /// eTypedefType::NamespaceExposition or
262 /// eTypedefType::DeclarationExposition. This primary
263 /// expression must not end with a semicolon, codegen
264 /// will insert a ';' if necessary.
265 string init{""};
266
267 _cx string Codegen() const {
268 switch (init_type) {
269 case eTypedefType::Typealias:
270 return "using " + ident + " = " + init + ";\n";
271 case eTypedefType::NamespaceExposition:
272 return "using namespace " + ident + ";\n";
273 case eTypedefType::DeclarationExposition:
274 return "using " + ident + ";\n";
275 default:
276 throw cxx::UnknownEnumEntry<eTypedefType>();
277 }
278 }
279};

281struct EnumEntry {
282 /// Name of the enum entry.
283 string name;
284 /// Optional default value for the enum entry.
285 optional<string> value;
286
287 /// @brief Constructor.
288 ///
289 /// @param name Name of the enum entry.
290 /// @param value Optional default value for the enum entry.
291 EnumEntry(const string& name, optional<string> value = std::nullopt)
292 : name(name), value(value) {}
293
294 /// @brief Code generation method.
295 ///

302 } else {
303 return name;
304 }

308/// Generates an enum declaration.
309struct EnumDecl {
310 string name;
311 /// Type of the enum. Unspecified if none.
312 string type;
313 /// If true, generate a scoped enum (enum class).
314 bool is_scoped =
315 false;
316 /// List of enumerators.
317 vector<EnumEntry> enumerators;
318 _cx string GetEnumDecl() const {
319 return (is_scoped ? "enum class " + name + " : " + type
320 : "enum " + name + " : " + type);
321 }
322
323 _cx string Codegen() const {
324 if (enumerators.empty())
325 return GetEnumDecl() + ";\n";
326 else
327 return GenerateExprList(enumerators, ",\n ", false,
328 GetEnumDecl() + " {\n ", "\n};\n\n");
329 }

367struct MethodDeclModifiers {
368 bool is_constexpr{false};
369 bool is_static{false};
370 bool is_inline{false};
371 bool is_noexcept{false};
372 bool is_const{false};
373
374 _cxin string GenPrefixMods() const {
375 string ret = (is_static ? "static " : "");
376 ret += (is_constexpr ? "constexpr " : "");
377 ret += (is_inline ? "inline " : "");
378 return ret;
379 }
380
381 _cxin string GenPostfixMods() const {
382 string ret = (is_const ? "const " : "");
383 ret += (is_noexcept ? "noexcept " : "");
384 return ret;
385 }

390struct MethodPostInitializer {
391 string member{};
392 string expr{};
393 string Codegen() const { return member + "(" + expr + ")"; }

395/// Represents a method declaration and optionally its definition.
396struct MethodDecl {
397 string name;
398 string return_type{"void"};
399 MethodDeclModifiers mods;
400 vector<MethodParameter> params;
401 optional<string> definition;
402 optional<TemplateSignature> template_signature;
403 optional<std::vector<string>> template_specialization;
404 /// When set, will make the method assigned to
405 /// string with '=' operator. Definition will be ignored.
406 optional<string> equal_to;
407 /// for constructors, generated when not empty.
408 vector<MethodPostInitializer>
409 post_initializer;
410
411 optional<string> comment_before;
412 optional<string> comment_after;
413 _cx string Codegen() const {
414 return GenerateExprList(
415 params, kCommaSeparator, false,
416 [=]() {
417 return comment_before.value_or("") +
418 (template_signature
419 ? template_signature.value().Codegen() + "\n"
420 : "") +
421 mods.GenPrefixMods() +
422 (template_specialization
423 ? (return_type + " " + name +
424 GenerateExprList(template_specialization.value(),
425 kCommaSeparator, false, "<", ">("))
426 : (return_type + " " + name + "("));
427 }(),
428 [=] {
429 string ret = ")" + mods.GenPostfixMods();
430 if (equal_to) {
431 ret += " = " + equal_to.value() + ";" + comment_after.value_or("") +
432 "\n\n";
433 return ret;
434 } else {
435 if (not post_initializer.empty())
436 ret += GenerateExprList(post_initializer, kCommaSeparator, false,
437 " : ");
438 if (definition)
439 ret += "{" + definition.value() + "}" +
440 comment_after.value_or("") + "\n\n";
441 else
442 ret += ";\n";
443 return ret;
444 }
445 }());
446 }
447};

449/// Generates a union declaration.
450struct UnionDecl {
451 struct UnionMemberVariant {
452 std::variant<std::reference_wrapper<const UnionMember>,
453 std::reference_wrapper<const MethodDecl>>
454 member;
455 string Codegen() const {
456 return std::visit([](const auto& m) { return m.get().Codegen(); },
457 member);
458 }
459 };
460
461 public:
462 string name;
463 vector<UnionMember> members;
464 vector<MethodDecl> methods;
465
466 _cx string Codegen() const {
467 if (members.empty())
468 return "union " + name + ";\n";
469 else {
470 std::vector<UnionMemberVariant> all_members;
471 all_members.reserve(members.size() + methods.size());
472 for (const auto& m : members) {
473 all_members.push_back(UnionMemberVariant{std::ref(m)});
474 }
475 for (const auto& m : methods) {
476 all_members.push_back(UnionMemberVariant{std::ref(m)});
477 }
478 return GenerateExprList(all_members, ";\n", true,
479 "union " + name + " {\n", "};\n", " ");
480 }
481 }
482};

484/// Represents a class declaration.
485struct ClassDecl {
486 string name;
487 optional<string> base_class;
488 vector<MethodDecl> methods;
489 vector<string>
490 member_variables;
491
492 _cx string Codegen() const {
493 string result = "class " + name;
494
495 if (base_class.has_value()) {
496 result += " : public " + base_class.value();
497 }
498
499 result += " {\npublic:\n";
500
501 // Generate member variables
502 for (const auto& member : member_variables) {
503 result += " " + member + ";\n";
504 }
505
506 // Generate methods
507 for (const auto& method : methods) {
508 result += " " + method.Codegen() + ";\n";
509 }
510
511 result += "};\n";
512
513 return result;
514 }
