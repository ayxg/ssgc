28// Class:{RtOperationalAllocator}
29// Brief:{Memory stack for a C& program instance.}
30// Detail: {
31// Represents the operational stack of a C& program instance.
32// The memory is encapsulated in a vector of RtVal objects.
33// Used to store the intermediate results of a program execution.
34//
35// Currently Implemented Operations:
36// - Top
37// - Top with an offset
38// - Push
39// - Pop
40// - PopN
41// - PopGet
42// - PushCopy
43// - Swap
44// - UnaryOperation
45// - BinaryOperation
46// - InPlaceBinaryOperation
47//
48// Good reference for the possible action this class must be able to perform:
49// - https://docs.python.org/3/library/dis.html
50// }
51//---------------------------------------------------------//
52class RtOperationalAllocator {
53 public:
54 // Get a ref to the top element of the stack.
55 RtVal& Top() { return memory_.back(); }
56 const RtVal& Top() const { return memory_.back(); }
57
58 // Top with an offset. Offset must be greater than 0.
59 RtVal& Top(size_t offset) {
60 assert(offset > 0);
61 return memory_[memory_.size() - offset--];
62 }
63
64 const RtVal& Top(size_t offset) const {
65 assert(offset > 0);
66 return memory_[memory_.size() - offset--];
67 }
68
69 // Push an element onto the stack.
70 void Push(RtVal&& rtv) { memory_.push_back(std::forward<RtVal>(rtv)); }
71
72 // Push an element onto the stack.
73 void Push(const RtVal& rtv) {
74 memory_.push_back(std::forward<const RtVal>(rtv));
75 }
76
77 // Removes the top-of-stack item
78 void Pop() { memory_.pop_back(); }
79
80 // Pop a number of items from the top of the stack.
81 void PopN(size_t n) {
82 while (n != 0) {
83 Pop();
84 n--;
85 }
86 }
87
88 // Removes the top-of-stack item and returns an rvalue reference
89 RtVal&& PopGet() {
90 RtVal val{std::move(memory_.back())};
91 memory_.pop_back();
92 return std::move(val);
93 }
94
95 // Push the i-th item of the stack to the top of the stack
96 // without removing it from its original location
97 // assert i > 0
98 // STACK.append(STACK[-i])
99 void PushCopy(size_t n) {
100 RtVal copy_rtv{Top(n)};
101 Push(std::move(copy_rtv));
102 }
103
104 // Swap the top of the stack with the i-th element:
105 // STACK[-i], STACK[-1] = STACK[-1], STACK[-i]
106 void Swap(size_t n) {
107 RtVal& val_to_move_to = Top(n);
108 RtVal val_to_move = Top();
109 Top() = val_to_move_to;
110 val_to_move_to = val_to_move;
111 }
112
113 // Unary operations take the top of the stack, apply the operation, and push
114 // the result back on the stack.
115 // Implements STACK[-1] = <UNARY_OPERATOR>STACK[-1].
116 void UnaryOperation(std::function<void(RtVal&)> op) { op(Top()); }
117
118 // Binary and in-place operations
119 // Binary operations remove the top two items from the stack(STACK[-1] and
120 // STACK[-2])
121 // They perform the operation, then put the result back on the stack.
122 void BinaryOperation(std::function<RtVal && (RtVal&&, RtVal&&)> op) {
123 RtVal&& rhs = PopGet();
124 RtVal&& lhs = PopGet();
125 Push(op(std::forward<RtVal>(lhs), std::forward<RtVal>(rhs)));
126 }
127
128 // In place operations are like binary operations, but the operation is done
129 // in-place when STACK[-2] supports it, and the resulting STACK[-1] may be
130 // (but does not have to be) the original STACK[-2].
131 // rhs = STACK.pop()
132 // lhs = STACK.pop()
133 // STACK.append(lhs op rhs)
134 void InPlaceBinaryOperation(std::function<void(RtVal&, RtVal&&)> op) {
135 RtVal&& rhs = PopGet();
136 RtVal& lhs = Top();
137 op(lhs, std::forward<RtVal>(rhs));
138 }
139
140 private:
141 std::vector<RtVal> memory_;
142};

147//---------------------------------------------------------//
148// Class:{RtOperationalAllocator}
149// Brief:{Memory stack for a C& program instance.}
150// Detail: {
151// Represents the static storage of a C& program instance.
152// Allocates memory for:
153// - Named declarations ( with a static storage specifier)
154// in method definitions.
155// - Global declarations, anything declared in the gloabl scope.
156// - Constants, anything declared with the const specifier.
157// - Variables of namespaces, or static variables of classes.
158// - Class instance variables.
159// }
160//---------------------------------------------------------//
161class RtStaticAllocator {
162 public:
163 using memory_address = std::list<RtVal>::iterator;
164 using rtval_ref = std::reference_wrapper<RtVal>;
165 struct memory_descriptor {
166 memory_address address;
167 size_t size;
168
169 RtVal& Deref() const { return *address; }
170 };
171
172 public:
173 // Allocate memory for a single new RtVal object
174 // which will have static lifetime duration.
175 // Memory will be alloc as long as program is running.
176 memory_descriptor AllocStatic() {
177 memory_.emplace_back();
178 return {.address = std::prev(memory_.end()), .size = 1};
179 }
180
181 // Alloc Constant
182 memory_descriptor AllocConstant(const RtVal& rtv) {
183 constants_.emplace(constants_.size(), new RtVal(rtv));
184 return {.address = std::prev(memory_.end()), .size = 1};
185 }
186
187 // Once allocated memory is only release all-at-once at end of program.
188 void ReleaseMemory() {
189 // Clear all maps
190 static_storage_.clear();
191 constants_.clear();
192
193 // Clear all memory
194 memory_.clear();
195 }
196
197 private:
198 std::list<RtVal> memory_;
199
200 // Memory for variables of namespaces, or static variables of classes.
201 std::unordered_map<memory_address, RtVal*> static_storage_;
202
203 // Memory for constants
204 std::unordered_map<memory_address, RtVal*> constants_;
205};
