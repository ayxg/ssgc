143// Section:{Native Type Concepts}
144// Brief:{
145// These cconcepts categorize the native types in the C& runtime. For use in
146// NativeVariant visitation and type checking.
147// }
148// Detail:{
149// iOneOfNativeLiterals: Type is one of the literal types, stored
150// directly as a value in the NativeVariant.
151// iOneOfNativeArithmetic: Type is one of the literal values which represents a
152// integral or real numeric value. This includes bool
153// and RtNone. Any operation appplied to none result in
154// none. Any operation applied with None results in the
155// input value.
156// }

158template <class T>
159concept iOneOfNativeLiterals =
160 mta::req::iOneOf<NativeIntT, NativeUnsignedT, NativeDoubleT, NativeBoolT,
161 NativeCharT, NativeByteT, NativeCStringT, NativeNoneT,
162 NativeUndefinedT, NativePtrT>;
163
164template <class T>
165concept iOneOfNativeArithmetic =
166 mta::req::iOneOf<NativeIntT, NativeUnsignedT, NativeDoubleT, NativeBoolT,
167 NativeCharT, NativeByteT, NativeNoneT>;
168template <class T>
169concept iOneOfNativeNonLiterals =
170 mta::req::iOneOf<NativeStringT, NativeNamespaceT, NativeObjectT,
171 NativeMethodT, NativeCppMethodT, NativeListT>;

180class RtVal {
181 public:
182 constexpr static inline RtVal NewNone() {
183 RtVal new_val;
184 new_val.data_ = NativeNoneT();
185 return new_val;
186 }
187
188 constexpr static inline RtVal NewUndefined() {
189 RtVal new_val;
190 new_val.data_ = NativeUndefinedT();
191 return new_val;
192 }
193
194 public:
195 constexpr size_t Idx() const { return data_.index(); }
196
197 constexpr eNativeTypeIndex::enum_type NativeIdx() const {
198 return static_cast<eNativeTypeIndex::enum_type>(data_.index());
199 }
200
201 constexpr NativeVariant& Variant() { return data_; }
202
203 constexpr inline void Copy(const NativeVariant& other) {
204 const auto xCopyRtVal = [this](const auto& other_val) constexpr {
205 using T = std::decay_t<decltype(other_val)>;
206 if constexpr (iOneOfNativeLiterals<T>) {
207 data_ = other_val;
208 } else if constexpr (iOneOfNativeNonLiterals<T>) {
209 data_ = std::make_unique<std::decay_t<T>::element_type>(*other_val);
210 } else {
211 throw std::logic_error("Unknown type in RtVal copy constructor.");
212 }
213 };
214 mta::visit_overloaded(other, xCopyRtVal);
215 }
216
217 constexpr inline void Copy(const RtVal& other) { Copy(other.data_); }
218
219 constexpr inline RtVal Clone() const {
220 RtVal new_val;
221 new_val.Copy(data_);
222 return new_val;
223 }
224
225 template <class T>
226 constexpr T& GetDataRef() {
227 return get<T>(data_);
228 }
229
230 template <class T>
231 constexpr const T& GetDataView() {
232 return get<T>(data_);
233 }
234
235 template <class T>
236 constexpr T GetDataCopy() {
237 return get<T>(value_);
238 }
239
240 public:
241 constexpr RtVal() = default;
242 constexpr RtVal(const RtVal& other) : RtVal(other.data_){};
243 constexpr RtVal(RtVal&& other) = default;
244 constexpr RtVal(const NativeVariant& other) { Copy(other); }
245 constexpr RtVal& operator=(const NativeVariant& other) {
246 Copy(other);
247 return *this;
248 }
249 constexpr RtVal& operator=(const RtVal& other) {
250 Copy(other);
251 return *this;
252 }
253
254 private:
255 NativeVariant data_;
256};

321// Class:{NamedRtValSet}
322// Brief:{
323// Container for a RtVal with a string_view name.
324// The name should not be modified after construction.
325// }
326//---------------------------------------------------------//
327
328class NamedRtValSet {
329 public:
330 using Set = std::unordered_set<NamedRtVal, NamedRtValHash, std::equal_to<>>;
331 using SetIter = Set::iterator;
332 using SetConstIter = Set::const_iterator;
333 using ExpectedRtValPtr = cxx::Expected<NamedRtVal::ValPtr>;
334
335 ExpectedRtValPtr GetRtValPtr(const string_view& name) {
336 if (data_.contains(name)) {
337 return ExpectedRtValPtr::Success(data_.find(name)->ValuePtr());
338 } else
339 return ExpectedRtValPtr::Failure(
340 string("[NamedRtValSet][GetRtVal] the name '") + name.data() +
341 "' does not exist in this set.");
342 }
343
344 ExpectedRtValPtr GetRtValPtr(size_t idx) {
345 if (idx < data_.size()) {
346 auto it = data_.begin();
347 std::advance(it, idx);
348 return ExpectedRtValPtr::Success(it->ValuePtr());
349 } else
350 return ExpectedRtValPtr::Failure(
351 string("[NamedRtValSet][GetRtVal] the index '") +
352 std::to_string(idx) + "' is out of bounds.");
353 }
354
355 // Returns a reference to the RtVal with the given name.
356 // !!Unchecked access!! If not found, undefined behavior.
357 RtVal& GetRtVal(const string_view& name) {
358 return *data_.find(name)->ValuePtr();
359 }
360
361 // Returns a reference to the RtVal with the given index.
362 // !!Unchecked access!! If not found, undefined behavior.
363 RtVal& GetRtVal(size_t idx) {
364 auto it = data_.begin();
365 std::advance(it, idx);
366 return *it->ValuePtr();
367 }
368
369 bool Contains(const string_view& name) const { return data_.contains(name); }
370
371 bool Contains(size_t idx) const { return idx < data_.size(); }
372
373 cxx::BoolError Emplace(NamedRtVal&& named_val) {
374 auto insertion_result = data_.emplace(forward<NamedRtVal>(named_val));
375 return insertion_result.second;
376 }
377
378 cxx::BoolError Emplace(const string_view& name, RtVal& value) {
379 return Emplace(NamedRtVal(name, value));
380 }
381
382 cxx::BoolError Erase(const string_view& name) {
383 if (Contains(name)) {
384 data_.erase(data_.find(name));
385 return true;
386 } else
387 return string("[NamedRtValSet][Erase] Name '") + name.data() +
388 "' does not exist in this set.";
389 }
390
391 cxx::BoolError Erase(size_t idx) {
392 if (Contains(idx)) {
393 auto it = data_.begin();
394 std::advance(it, idx);
395 data_.erase(it);
396 return true;
397 } else
398 return string("[NamedRtValSet][Erase] Index '") + std::to_string(idx) +
399 "' is out of bounds.";
400 }
401
402 cxx::BoolError PopBack() {
403 if (data_.empty()) {
404 return "[NamedRtValSet][PopBack] Set is empty.";
405 } else {
406 data_.erase(std::prev(data_.end()));
407 return true;
408 }
409 }
410
411 // !!Unchecked access!! If set is empty, undefined behavior.
412 const NamedRtVal& Back() const { return *std::prev(data_.end()); }
413
414 void Clear() { data_.clear(); }
415
416 public:
417 NamedRtValSet() = default;
418 NamedRtValSet(const NamedRtValSet&) = default;
419 NamedRtValSet(NamedRtValSet&&) = default;
420 NamedRtValSet& operator=(const NamedRtValSet&) = default;
421 NamedRtValSet& operator=(NamedRtValSet&&) = default;
422
423 // Utility constructor for creating a null set from a list of names.
424 NamedRtValSet(mta::req::iConvertibleTo<string_view> auto... names)
425 : data_(NamedRtVal{names}...) {
426 assert(mta::are_unique(names...) &&
427 "[NamedRtValSet][NamedRtValSet] List of names passed to constructor "
428 "must be unique.");
429 }
430
431 private:
432 Set data_;
433};

469// Class:{RtDynNamespace}
470// Brief:{ Represents a static namespace, object definition or
471// the global namespace in the C& runtime.}
472// Detail:{
473// NOTE: Dyn namespace type enumation is named 'eCategories'
474// to avoid confusion with types.
475//
476// A namespace in C& can be one of four types:
477// - Undefined namespace, always named '#',
478// an undefined or invalid namespace/class def.
479// '#' can never be a namespace name, here its used to
480// signify the null namespace.
481//
482// - A static namespace, containing static members only.
483// Any re-definition of this namespace will add to
484// the existing namespace.
485//
486// - A class definition which is a static namespace,
487// containing a default generated _init_ method
488// which describes the contruction of the non-static
489// members of a RtDynObject copied from the class def's
490// instance members.
491// Contains an _rtti_ value which is the unique runtime
492// type index of this class.
493// For now 0 will represent the null type, or lack-of-type.
494// Any re-definition of this class namespace is an error.
495//
496// - The global namespace of this program instance. Same
497// as a static namespace with special optimizations.
498// The name of the global namespace should be '_global_'.
499//
500// A namespace may have a parent namespace, which is called
501// to when resolving a name. The global namespace has no parent.
502// All other namespaces have a parent or are children of the
503// global namespace.
504// Upon querying the pare of global or undefined namespaces,
505// the same namespace is returned.
506// }
507//---------------------------------------------------------//
508class RtDynNamespace {
509 public:
510 using ExpectedRtValPtr = NamedRtValSet::ExpectedRtValPtr;
511
512 enum class eCategory { kUndefined, kNamespace, kClass, kGlobal };
513
514 // !!DO NOT MODIFY or ACCESS!!
515 // Static null namespace used by default constructor of RtDynObject
516 static RtDynNamespace gStaticUndefinedNamespace;
517
518 public:
519 constexpr eCategory Category() const { return category_; }
520 constexpr size_t Rtti() const { return rtti_; }
521 constexpr string_view Name() const { return name_; }
522
523 // !!Warning!!
524 // If namespace or class def has a nullptr parent, undefined behavior.
525 constexpr RtDynNamespace& Parent() {
526 if (IsUndefined() || IsGlobal()) return *this;
527 return *parent_;
528 }
529 constexpr const RtDynNamespace& Parent() const {
530 if (IsUndefined() || IsGlobal()) return *this;
531 return *parent_;
532 }
533 constexpr bool IsUndefined() const {
534 return category_ == eCategory::kUndefined;
535 }
536 constexpr bool IsNamespace() const {
537 return category_ == eCategory::kNamespace;
538 }
539 constexpr bool IsClass() const { return category_ == eCategory::kClass; }
540 constexpr bool IsGlobal() const { return category_ == eCategory::kGlobal; }
541
542 constexpr bool AddInstanceDecl(const NamedRtVal& named_val) {
543 return instance_decls_.Emplace(NamedRtVal(named_val));
544 }
545
546 constexpr bool AddInstanceDecl(NamedRtVal&& named_val) {
547 return instance_decls_.Emplace(forward<NamedRtVal>(named_val));
548 }
549
550 constexpr bool AddInstanceDecl(const string_view& name, RtVal& value) {
551 return instance_decls_.Emplace(NamedRtVal(name, value));
552 }
553
554 constexpr bool AddDecl(const NamedRtVal& named_val) {
555 return decls_.Emplace(NamedRtVal(named_val));
556 }
557
558 constexpr bool AddDecl(NamedRtVal&& named_val) {
559 return decls_.Emplace(forward<NamedRtVal>(named_val));
560 }
561
562 constexpr bool AddDecl(const string_view& name, RtVal& value) {
563 return decls_.Emplace(NamedRtVal(name, value));
564 }
565
566 public:
567 constexpr const NamedRtValSet& InstanceDecls() const {
568 return instance_decls_;
569 }
570
571 constexpr bool ContainsLocal(const string_view& name) const {
572 return decls_.Contains(name);
573 }
574
575 constexpr ExpectedRtValPtr ResolveLocal(const string_view& name) {
576 return decls_.GetRtValPtr(name);
577 }
578
579 // Faster than regular resolve but undefined name does not exist.
580 constexpr RtVal& ResolveLocalUnchecked(const string_view& name) {
581 return decls_.GetRtVal(name);
582 }
583
584 constexpr bool Contains(const string_view& name) const {
585 if (ContainsLocal(name)) return true;
586 if (IsGlobal()) return false;
587 return Parent().Contains(name);
588 }
589
590 // It is assumed that this is never called on an undefined namespace.
591 // It is assumed that all static namespace and class defs have a parent.
592 constexpr ExpectedRtValPtr Resolve(const string_view& name) {
593 if (ContainsLocal(name)) {
594 return ResolveLocal(name);
595 } else {
596 if (IsGlobal())
597 return ExpectedRtValPtr::Failure(
598 string("Global namespace does not contain the variable named '") +
599 name.data() + "'.");
600 return Parent().Resolve(name);
601 }
602 }
603
604 // !!Unchecked access!! If not found, undefined behavior.
605 // Use Contains() to check if the name exists in this hierarchy.
606 // Checks all namespaces in the hierarchy.
607 constexpr RtVal& ResolveUnchecked(const string_view& name) {
608 if (ContainsLocal(name)) {
609 return ResolveLocalUnchecked(name);
610 } else {
611 return Parent().ResolveUnchecked(name);
612 }
613 }
614
615 public:
616 RtDynNamespace() = default;
617 RtDynNamespace(const RtDynNamespace&) = default;
618 RtDynNamespace(RtDynNamespace&&) = default;
619 RtDynNamespace& operator=(const RtDynNamespace&) = default;
620 RtDynNamespace& operator=(RtDynNamespace&&) = default;
621
622 RtDynNamespace(const string_view& name)
623 : category_(eCategory::kNamespace), name_(name) {}
624
625 private:
626 eCategory category_{eCategory::kUndefined};
627 size_t rtti_{kUndefinedRtti};
628 RtDynNamespace* parent_{nullptr};
629 string_view name_{kUndefinedNamespaceName};
630 NamedRtValSet decls_; // Declarations in this namespace.
631 NamedRtValSet instance_decls_; // non-static members.
632 // copied over for each instance of a class.
633};

683// Class:{RtDynMethod}
684// Brief:{
685// Represents a dynamic method in the C& runtime.
686// A method is a set of instructions which can be executed.
687// The instructions may be any C++ void method which
688// takes a RtDynNamespace& argument - but it is
689// Expected that the method is a bound IrCodeAction which will
690// execute an action on the runtime memory stack. Resolving
691// the variable values from the currently bound context.
692// }
693//---------------------------------------------------------//
694class RtDynMethod {
695 public:
696 constexpr const string_view& Name() const { return name_; }
697 constexpr const NamedRtValSet& Arguments() const { return arguments_; }
698 constexpr const RtDynNamespace& Context() const { return context_; }
699 void Call() {
700 for (auto& instruction : instructions_) {
701 instruction(context_);
702 }
703 }
704
705 private:
706 string_view name_{""};
