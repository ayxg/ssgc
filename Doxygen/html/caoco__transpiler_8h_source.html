173int Transpile(const irast::Program& program, std::ostream& out) {
174 EnterProgramDefinition(program, out);
175 IncludeCandStandardLibraries(program, out);
176 IncludePredefinedCppLibraries(program, out);
177 IncludeUserDefinedCppFiles(program, out);
178 EnterGlobalNamespace(program, out);
179 for (const irast::Statement& stmt : program.statements) {
180 std::visit([&](auto&& arg) { TranspilePragmaticStmt(arg, out); }, stmt);
181 }
182 ExitGlobalNamespace(program, out);
183 ExitProgramDefinition(program, out);
184}
