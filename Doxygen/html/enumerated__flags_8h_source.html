18/// @brief Represents a set of flags that can be combined using bitwise OR.
19/// @tparam EnumT An enumeration type which will be used as flags.
20/// @tparam UnderlyingT Defaults to the underlying type of EnumT.
21///
22/// Flags may be appended or removed. Retrieve the flags using Get().
23template <mta::req::iEnumeration EnumT,
24 mta::req::iIntegral UnderlyingT = std::underlying_type_t<EnumT>>
25class EnumeratedFlags {
26 public:
27 /// Retrieve the flags by value.
28 UnderlyingT Get() const { return flags_; }
29
30 /// Append a single flag.
31 constexpr void Append(EnumT flags) { flags_ |= flags; }
32
33 /// Append a list of flags, applied using bitwise OR.
34 constexpr void Append(EnumT flags, std::same_as<EnumT> auto... other_flags) {
35 (..., (flags_ |= other_flags));
36 }
37
38 /// Remove a single flag.
39 constexpr void Remove(EnumT flags) { flags_ &= ~(flags); }
40
41 /// Remove a list of flags, applied using bitwise AND with bitwise NOT.
42 constexpr void Remove(EnumT flags, std::same_as<EnumT> auto... other_flags) {
43 flags_ &= ~(flags);
44 (..., (flags_ &= ~(other_flags)));
45 }
46
47 /// Bitwise OR assignment operator.
48 constexpr EnumeratedFlags& operator|=(EnumT flag) {
49 flags_ |= flag;
50 return *this;
51 }
52
53 /// Default value is 0.
54 EnumeratedFlags() = default;
55
56 /// Construct from a single flag.
57 EnumeratedFlags(EnumT flags) : flags_(0) { flags_ |= flags; }
58
59 /// Construct from a list of flags which will be combined using bitwise OR.
60 EnumeratedFlags(EnumT flags, std::same_as<EnumT> auto... other_flags)
61 : flags_(flags) {
62 (..., (flags_ |= other_flags));
63 }
64
65 /// Implicit conversion to the enumeration type.
66 constexpr operator EnumT() const { return static_cast<EnumT>(flags_); }
67
68 /// Implicit conversion to the underlying type.
69 constexpr operator UnderlyingT() const { return flags_; }
70
71 private:
72 UnderlyingT flags_{0};
73};
74}; // namespace cxx
